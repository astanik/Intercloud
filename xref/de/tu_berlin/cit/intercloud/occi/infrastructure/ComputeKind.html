17  package de.tu_berlin.cit.intercloud.occi.infrastructure;
18  
19  import de.tu_berlin.cit.intercloud.occi.core.annotations.Attribute;
20  import de.tu_berlin.cit.intercloud.occi.core.annotations.Category;
21  import de.tu_berlin.cit.intercloud.occi.core.annotations.Kind;
22  import de.tu_berlin.cit.intercloud.occi.core.annotations.Attribute.AttributeType;
23  
24  
25  @Kind(schema = ComputeKind.ComputeSchema, term = ComputeKind.ComputeTerm)
26  public class ComputeKind extends Category {
27  
28  	public final static String ComputeTitle = "Compute Resource";
29  	
30  	public final static String ComputeSchema = "http://schema.ogf.org/occi/infrastructure#";
31  	
32  	public final static String ComputeTerm = "compute";
33  	
34  	public final static String NetworkInterfaceTerm = "networkinterface";
35  	
36  	public ComputeKind() {
37  		super(ComputeTitle);
38  	}
39  
40  	public ComputeKind(String title) {
41  		super(title);
42  	}
43  
44  	public enum Architecture {
45  		x86,
46  		x64
47  	}
48  	
49  	@Attribute(name = "occi.compute.architecture",
50  			type = AttributeType.ENUM,
51  			mutable = true,
52  			description = "CPU Architecture of the instance: Enum{x86,x64}")
53  	public Architecture architecture = null;
54  
55  	@Attribute(name = "occi.compute.cores",
56  			type = AttributeType.INTEGER,
57  			mutable = true,
58  			value = "1",
59  			description = "Number of virtual CPU cores assigned to the instance")
60  	public Integer cores = 1;
61  
62  	@Attribute(name = "occi.compute.hostname",
63  			type = AttributeType.STRING,
64  			mutable = true,
65  			description = "Fully qualified DNS host name for the instance")
66  	public String hostname = null;
67  
68  	@Attribute(name = "occi.compute.share",
69  			type = AttributeType.INTEGER,
70  			mutable = true,
71  			description = "Relative number of CPU shares for the instance")
72  	public Integer share = null;
73  
74  	@Attribute(name = "occi.compute.memory",
75  			type = AttributeType.DOUBLE,
76  			mutable = true,
77  			description = "Minimum RAM in gigabytes allocated to the instance")
78  	public Double memory = null;
79  
80  	public enum State {
81  		active, 
82  		inactive, 
83  		suspended, 
84  		error
85  	}
86  	
87  	@Attribute(name = "occi.compute.state",
88  			type = AttributeType.ENUM,
89  			required = true,
90  			description = "Current state of the instance: Enum{active, inactive, suspended, error}")
91  	public State state = null;
92  
93  	@Attribute(name = "occi.compute.message",
94  			type = AttributeType.STRING,
95  			description = "Human-readable explanation of the current instance state")
96  	public String message = null;
97  
98  }
