17  package de.tu_berlin.cit.intercloud.occi.servicecatalog;
18  
19  import de.tu_berlin.cit.intercloud.occi.core.annotations.Attribute;
20  import de.tu_berlin.cit.intercloud.occi.core.annotations.Category;
21  import de.tu_berlin.cit.intercloud.occi.core.annotations.Mixin;
22  import de.tu_berlin.cit.intercloud.occi.core.annotations.Attribute.AttributeType;
23  
24  @Mixin(schema = ServiceCatalogMixin.CatalogSchema, term = ServiceCatalogMixin.CatalogTerm)
25  public class ServiceCatalogMixin extends Category {
26  
27  	public final static String CatalogTitle = "Service Catalog Resource";
28  	
29  	public final static String CatalogSchema = "http://schema.cit.tu-berlin.de/occi/catalog#";
30  	
31  	public final static String CatalogTerm = "service";
32  	
33  	public ServiceCatalogMixin() {
34  		super(CatalogTitle);
35  	}
36  
37  	public ServiceCatalogMixin(String title) {
38  		super(title);
39  	}
40  
41  	@Attribute(name = "occi.catalog.price",
42  			type = AttributeType.DOUBLE,
43  			mutable = true,
44  			required = true,
45  			description = "The price for the described service")
46  	public double price;
47  
48  	@Attribute(name = "occi.catalog.currency",
49  			type = AttributeType.STRING,
50  			mutable = true,
51  			required = true,
52  			value = "Euro",
53  			description = "The currency of the price")
54  	public String currency;
55  
56  	@Attribute(name = "occi.catalog.billingincrements",
57  			type = AttributeType.STRING,
58  			mutable = true,
59  			required = true,
60  			value = "per hour",
61  			description = "The billing increments of the price, e.g. per GB, per hour, per second, etc.")
62  	public String billingIncrements;
63  	
64  
65  }
