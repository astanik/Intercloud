17  package de.tu_berlin.cit.intercloud.occi.core;
18  
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  import de.tu_berlin.cit.intercloud.occi.core.annotations.Summary;
23  import de.tu_berlin.cit.intercloud.occi.core.incarnation.ClassificationRegistry;
24  import de.tu_berlin.cit.intercloud.occi.core.xml.classification.ClassificationDocument;
25  import de.tu_berlin.cit.intercloud.xmpp.rest.ResourceInstance;
26  import de.tu_berlin.cit.intercloud.xmpp.rest.XwadlBuilder;
27  import de.tu_berlin.cit.intercloud.xmpp.rest.annotations.XmppMethod;
28  import de.tu_berlin.cit.intercloud.xmpp.rest.xwadl.ResourceTypeDocument;
29  import de.tu_berlin.cit.intercloud.xmpp.rest.xwadl.ResourceTypeDocument.ResourceType;
30  
31  public class OcciXwadlBuilder extends XwadlBuilder {
32  
33  	protected final static Logger logger = LoggerFactory.getLogger(OcciXwadlBuilder.class);
34  
35  	public static ResourceTypeDocument build(String path,
36  			ResourceInstance instance) {
37  		logger.info("Start building xwadl document ...");
38  		// create new document 
39  		ResourceTypeDocument xwadl = ResourceTypeDocument.Factory.newInstance();
40  		// set resource path
41  		ResourceType resType = xwadl.addNewResourceType();
42  		resType.setPath(path);
43  		logger.info("resource path=" + path);
44  		// check summary annotation
45  		if (instance.getClass().isAnnotationPresent(Summary.class)) {
46  			String summary = instance.getClass().getAnnotation(Summary.class).value();
47  			resType.addNewDocumentation().setTitle("Summary");
48  			resType.getDocumentation().setStringValue(summary);
49  		}
50  		// add occi classification grammar
51  		if(instance instanceof Collection || instance instanceof Resource ||
52  				instance instanceof Link) {
53  			ClassificationDocument classDoc = ClassificationRegistry.buildClassification(instance);
54  			resType.addNewGrammars().set(classDoc);
55  		}
56  		
57  		// search methods
58  		for(java.lang.reflect.Method method : instance.getClass().getMethods()) {
59  			// create method entry
60  			if(method.isAnnotationPresent(XmppMethod.class))
61  				XwadlBuilder.createMethodXWADL(method, resType.addNewMethod());
62  		}
63  		// TODO Actions
64  		
65  		logger.info("Finished building xwadl document: " + xwadl.toString());
66  		return xwadl;
67  	}
68  
69  }
