39  public final class ClassificationRegistry {
40  
41  	private final static Logger logger = LoggerFactory
42  			.getLogger(ClassificationRegistry.class);
43  
44  	private static ClassificationRegistry singleton = new ClassificationRegistry();
45  
46  	private final Map<String, Class<? extends Category>> classMapping;
47  
48  	/**
49  	 * A private Constructor prevents any other class from instantiating.
50  	 */
51  	private ClassificationRegistry() {
52  		logger.info("Retrieving namespace constants ...");
53  		this.classMapping = NamespaceConstants.mapping;
54  	}
55  
56  	/**
57  	 * Static 'instance' method
58  	 */
59  	public static ClassificationRegistry getInstance() {
60  		return singleton;
61  	}

67  	public static ClassificationDocument buildClassification(ResourceInstance instance) {
68  		logger.info("Start building " + instance.getClass().getSimpleName()
69  				+ " classification ...");
70  		ClassificationDocument doc = ClassificationDocument.Factory
71  				.newInstance();
72  
73  		doc.addNewClassification();
74  		if (instance.getClass().isAnnotationPresent(Classification.class)) {
75  			Classification classification = instance.getClass().getAnnotation(
76  					Classification.class);
77  			appendKindClassification(doc, classification);
78  			appendMixinClassification(doc, classification);
79  			appendLinkClassification(doc, classification);
80  		}
81  
82  		logger.info("Classification document has been build: ");
83  		logger.info(doc.toString());
84  		return doc;
85  	}
86  
87  	private static void appendLinkClassification(ClassificationDocument doc,
88  			Classification classification) {
89  		Class<? extends Category>[] links = classification.links();
90  
91  		// append links
92  		for (int i = 0; i < links.length; i++) {
93  
94  			if (!links[i].isAnnotationPresent(Link.class)) {
95  				throw new RuntimeException("the class is not a link");
96  			}
97  
98  			// create link element
99  			LinkClassification docLink = doc.getClassification()
100 					.addNewLinkType();
101 
102 			docLink.setSchema(links[i].getAnnotation(Link.class).schema());
103 			docLink.setTerm(links[i].getAnnotation(Link.class).term());
104 			docLink.setRelation(links[i].getAnnotation(Link.class).relation());
105 			// TODO: add optional title
106 
107 			appendAttributeClassification(docLink, links[i]);
108 		}
109 	}
110 
111 	private static void appendMixinClassification(ClassificationDocument doc,
112 			Classification classification) {
113 		Class<? extends Category>[] mixins = classification.mixins();
114 
115 		for (int i = 0; i < mixins.length; i++) {
116 
117 			if (!mixins[i].isAnnotationPresent(Mixin.class)) {
118 				throw new RuntimeException("the class is not a mixin");
119 			}
120 
121 			// create mixin element
122 			MixinClassification docMixin = doc.getClassification()
123 					.addNewMixinType();
124 
125 			docMixin.setSchema(mixins[i].getAnnotation(Mixin.class).schema());
126 			docMixin.setTerm(mixins[i].getAnnotation(Mixin.class).term());
127 			docMixin.setAppliesArray(mixins[i].getAnnotation(Mixin.class)
128 					.applies());
129 			// TODO: add optional title
130 
131 			appendAttributeClassification(docMixin, mixins[i]);
132 		}
133 	}
134 
135 	private static void appendKindClassification(ClassificationDocument doc,
136 			Classification classification) {
137 		Class<? extends Category> kind = classification.kind();
138 		if (kind.equals(Category.class)) {
139 			// no kind is set
140 			// this is not an exception!
141 			return;
142 		}
143 
144 		if (!kind.isAnnotationPresent(Kind.class)) {
145 			// the class is no kind
146 			throw new RuntimeException("the class is not a kind");
147 		}
148 
149 		// create kind element
150 		CategoryClassification docKind = doc.getClassification()
151 				.addNewKindType();
152 
153 		docKind.setSchema(kind.getAnnotation(Kind.class).schema());
154 		docKind.setTerm(kind.getAnnotation(Kind.class).term());
155 		// TODO: add optional title
156 
157 		appendAttributeClassification(docKind, kind);
158 	}
159 
160 	private static void appendAttributeClassification(CategoryClassification docKind,
161 			Class<? extends Category> kind) {
162 
163 		for (Field field : kind.getFields()) {
164 			if (field.isAnnotationPresent(Attribute.class)) {
165 				Attribute attribute = field.getAnnotation(Attribute.class);
166 				AttributeClassification attXml = docKind
167 						.addNewAttributeClassification();
168 				attXml.setName(attribute.name());
169 				attXml.setType(AttributeType.Enum.forString(attribute.type()
170 						.toString()));
171 				attXml.setMutable(attribute.mutable());
172 				attXml.setRequired(attribute.required());
173 				attXml.setDefault(attribute.value());
174 				attXml.setDescription(attribute.description());
175 			}
176 		}
177 	}
178 
179 	public Map<String, Class<? extends Category>> getClassMapping() {
180 		return this.classMapping;
181 	}
182 
183 	public void addClassMapping(String key, Class<? extends Category> value) {
184 		this.classMapping.put(key, value);
185 	}
186 }
