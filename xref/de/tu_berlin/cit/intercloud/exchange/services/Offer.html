17  package de.tu_berlin.cit.intercloud.exchange.services;
18  
19  import java.net.URISyntaxException;
20  
21  import de.tu_berlin.cit.intercloud.occi.core.Collection;
22  import de.tu_berlin.cit.intercloud.occi.core.OcciXml;
23  import de.tu_berlin.cit.intercloud.occi.core.annotations.Classification;
24  import de.tu_berlin.cit.intercloud.occi.core.annotations.Summary;
25  import de.tu_berlin.cit.intercloud.occi.sla.OfferKind;
26  import de.tu_berlin.cit.intercloud.xmpp.rest.annotations.Consumes;
27  import de.tu_berlin.cit.intercloud.xmpp.rest.annotations.Path;
28  import de.tu_berlin.cit.intercloud.xmpp.rest.annotations.Produces;
29  import de.tu_berlin.cit.intercloud.xmpp.rest.annotations.XmppMethod;
30  import de.tu_berlin.cit.intercloud.xmpp.rest.representations.UriText;
31  
32  @Path("/manager")
33  @Summary("This resource allows for manage "
34  		+ "SLA offer creation and agreement negotiation.")
35  @Classification(kind = OfferKind.class)
36  public class Offer extends Collection {
37  
38  	public Offer() {
39  		super();
40  	}
41  	
42  	@XmppMethod(XmppMethod.POST)
43      @Consumes(value = OcciXml.MEDIA_TYPE, serializer = OcciXml.class)
44      @Produces(value = UriText.MEDIA_TYPE, serializer = UriText.class)
45  	public UriText createOffer(OcciXml offerXml) {
46  		// create an offer and return its uri
47  		OfferInstance offer = new OfferInstance(offerXml);
48  		String path = this.addResource(offer);
49  		try {
50  			UriText uri = new UriText(path);
51  			return uri;
52  		} catch (URISyntaxException e) {
53  			// TODO Auto-generated catch block
54  			e.printStackTrace();
55  			return new UriText(); 
56  		}
57  	}
58  
59  }
