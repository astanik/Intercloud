18  package de.tu_berlin.cit.intercloud.xmpp.core.component;
19  
20  import de.tu_berlin.cit.intercloud.xmpp.core.packet.IQ;
21  import de.tu_berlin.cit.intercloud.xmpp.core.packet.Packet;
22  
23  
24  /**
25   * Manages components.
26   *
27   * @see Component
28   * @author Matt Tucker
29   */
30  public interface ComponentManager {
31  
32      /**
33       * Adds a component. The 
34       * method will be called on the component. The subdomain specifies the address of
35       * the component on a server. For example, if the subdomain is "test" and the XMPP
36       * server is at "example.com", then the component's address would be "test.example.com".
37       *
38       * @param subdomain the subdomain of the component's address.
39       * @param component the component.
40       * @throws ComponentException if the component connection is lost and the component cannot be added.
41       */
42      public void addComponent(String subdomain, Component component) throws ComponentException;
43  
44      /**
45       * Removes a component. The method will be called on the
46       * component.
47       *
48       * @param subdomain the subdomain of the component's address.
49       * @throws ComponentException if the component connection is lost and the component cannot be removed.
50       */
51      public void removeComponent(String subdomain) throws ComponentException;
52  
53      /**
54       * Sends a packet to the XMPP server. The "from" value of the packet must not be null.
55       * An <tt>IllegalArgumentException</tt> will be thrown when the "from" value is null.<p>
56       *
57       * Components are trusted by the server and may use any value in from address. Usually
58       * the from address uses the component's address as the domain but this is not required.
59       *
60       * @param component the component sending the packet.
61       * @param packet the packet to send.
62       * @throws ComponentException if the component connection is lost or unavialble during the time of sending and
63       * recieving packets.
64       */
65      public void sendPacket(Component component, Packet packet) throws ComponentException;
66  
67      /**
68       * Sends an IQ packet to the XMPP server and waits to get an IQ of type result or error.
69       * The "from" value of the packet must not be null. An <tt>IllegalArgumentException</tt>
70       * will be thrown when the "from" value is null.<p>
71       *
72       * If no answer is received from the server before the specified timeout then <tt>null</tt> will be returned.
73       *
74       * Components are trusted by the server and may use any value in from address. Usually
75       * the from address uses the component's address as the domain but this is not required.
76       *
77       * @param component the component sending the packet.
78       * @param packet the IQ packet to send.
79       * @param timeout the number of milliseconds to wait before returning an IQ error.
80       * @return the answer sent by the server. The answer could be an IQ of type result or
81       *         error. <tt>null</tt> will be returned if there is no response from the server.
82       * @throws ComponentException if the component connection is lost or unavialble during the time of sending and
83       * recieving packets.
84       */
85      public IQ query(Component component, IQ packet, long timeout) throws ComponentException;
86  
87      /**
88       * Sends an IQ packet to the server and returns immediately. The specified IQResultListener
89       * will be invoked when an answer is received.
90       *
91       * @param component the component sending the packet.
92       * @param packet the IQ packet to send.
93       * @param listener the listener that will be invoked when an answer is received.
94       * @throws ComponentException if the component connection is lost or unavialble during the time of sending and
95       * recieving packets.
96       */
97      public void query(Component component, IQ packet, IQResultListener listener) throws ComponentException;
98  
99      /**
100      * Returns a property value specified by name. Properties can be used by
101      * components to store configuration data. It is recommended that each
102      * component qualify property names to prevent overlap. For example a
103      * component that broadcasts messages to groups of users, might prepend
104      * all property names it uses with "broadcast.".
105      *
106      * @param name the property name.
107      * @return the property value.
108      */
109     public String getProperty(String name);
110 
111     /**
112      * Sets a property value. Properties can be used by components to
113      * store configuration data. It is recommended that each component
114      * qualify property names to prevent overlap. For example a component
115      * that broadcasts messages to groups of users, might prepend all
116      * property names it uses with "broadcast.".
117      *
118      * @param name the property name.
119      * @param value the property value.
120      */
121     public void setProperty(String name, String value);
122 
123     /**
124      * Returns the domain of the XMPP server. The domain name may be the IP address or the host
125      * name.
126      *
127      * @return the domain of the XMPP server.
128      */
129     public String getServerName();
130     
131     /**
132      * Returns true if components managed by this component manager are external
133      * components connected to the server over a network connection. Otherwise,
134      * the components are internal to the server.
135      *
136      * @return true if the managed components are external components.
137      */
138     public boolean isExternalMode();
139 
140 	public Object getLog();
141 }
