20  import org.dom4j.Element;

22  import java.util.ArrayList;
23  import java.util.Iterator;

26  /**
27   * Represents a field of a form. The field could be used to represent a question to complete,
28   * a completed question or a data returned from a search. The exact interpretation of the field
29   * depends on the context where the field is used.
30   *
31   * @author Gaston Dombiak
32   */
33  public class FormField {
34  
35      private Element element;
36  
37      FormField(Element element) {
38          this.element = element;
39      }
40  
41      /**
42       * Adds a default value to the question if the question is part of a form to fill out.
43       * Otherwise, adds an answered value to the question.
44       * 
45       * Nothing will be added if the provided argument is <tt>null</tt>.
46       * 
47       * @param value a default value or an answered value of the question.
48       */
49      public void addValue(Object value) {
50      	if (value == null) {
51      		return;
52      	}
53          element.addElement("value").setText(DataForm.encode(value));
54      }
55  
56      /**
57       * Removes all the values of the field.
58       */
59      @SuppressWarnings("unchecked")
60      public void clearValues() {
61          for (Iterator<Element> it = element.elementIterator("value"); it.hasNext();) {
62              it.next();
63              it.remove();
64          }
65      }
66  
67      /**
68       * Adds an available option to the question that the user has in order to answer
69       * the question.
70       * 
71       * If argument 'value' is <tt>null</tt> or an empty String, no option element 
72       * will be added.
73       * 
74       * @param label a label that represents the option. Optional argument.
75       * @param value the value of the option.
76       */
77      public void addOption(String label, String value) {
78      	if (value == null || value.trim().length() == 0) {
79      		return;
80      	}
81      	
82          Element option = element.addElement("option");
83          option.addAttribute("label", label);
84          option.addElement("value").setText(value);
85      }
86  
87      /**
88       * Returns the available options to answer for this question. The returned options cannot
89       * be modified but they will be updated if the underlying DOM object gets updated.
90       *
91       * @return the available options to answer for this question.
92       */
93      @SuppressWarnings("unchecked")
94      public List<Option> getOptions() {
95          List<Option> answer = new ArrayList<Option>();
96          for (Iterator<Element> it = element.elementIterator("option"); it.hasNext();) {
97              answer.add(new Option(it.next()));
98          }
99          return answer;
100     }
101 
102     /**
103      * Sets an indicative of the format for the data to answer. Valid formats are:
104      * 
105      * <ul>
106      * <li>text-single -> single line or word of text
107      * <li>text-private -> instead of showing the user what they typed, you show ***** to
108      * protect it
109      * <li>text-multi -> multiple lines of text entry
110      * <li>list-single -> given a list of choices, pick one
111      * <li>list-multi -> given a list of choices, pick one or more
112      * <li>boolean -> 0 or 1, true or false, yes or no. Default value is 0
113      * <li>fixed -> fixed for putting in text to show sections, or just advertise your web
114      * site in the middle of the form
115      * <li>hidden -> is not given to the user at all, but returned with the questionnaire
116      * <li>jid-single -> Jabber ID - choosing a JID from your roster, and entering one based
117      * on the rules for a JID.
118      * <li>jid-multi -> multiple entries for JIDs
119      * </ul>
120      *
121      * @param type an indicative of the format for the data to answer.
122      */
123     public void setType(Type type) {
124         element.addAttribute("type", type==null?null:type.toXMPP());
125     }
126 
127     /**
128      * Sets the attribute that uniquely identifies the field in the context of the form. If the
129      * field is of type "fixed" then the variable is optional.
130      *
131      * @param var the unique identifier of the field in the context of the form.
132      */
133     public void setVariable(String var) {
134         element.addAttribute("var", var);
135     }
136 
137     /**
138      * Sets the label of the question which should give enough information to the user to
139      * fill out the form.
140      *
141      * @param label the label of the question.
142      */
143     public void setLabel(String label) {
144         element.addAttribute("label", label);
145     }
146 
147     /**
148      * Sets if the question must be answered in order to complete the questionnaire.
149      *
150      * @param required if the question must be answered in order to complete the questionnaire.
151      */
152     public void setRequired(boolean required) {
153         // Remove an existing desc element.
154         if (element.element("required") != null) {
155             element.remove(element.element("required"));
156         }
157         if (required) {
158             element.addElement("required");
159         }
160     }
161 
162     /**
163      * Sets a description that provides extra clarification about the question. This information
164      * could be presented to the user either in tool-tip, help button, or as a section of text
165      * before the question.
166      * 
167      * If the question is of type FIXED then the description should remain empty.
168      * 
169      * No new description will be set, if the provided argument is <tt>null</tt> or an empty 
170      * String (although an existing description will be removed).
171      * 
172      * @param description provides extra clarification about the question.
173      */
174     public void setDescription(String description) {
175         // Remove an existing desc element.
176         if (element.element("desc") != null) {
177             element.remove(element.element("desc"));
178         }
179         
180         if (description == null || description.trim().length() == 0) {
181         	return;
182         }
183         
184         element.addElement("desc").setText(description);
185     }
186 
187     /**
188      * Returns true if the question must be answered in order to complete the questionnaire.
189      *
190      * @return true if the question must be answered in order to complete the questionnaire.
191      */
192     public boolean isRequired() {
193         return element.element("required") != null;
194     }
195 
196     /**
197      * Returns the variable name that the question is filling out.
198      *
199      * @return the variable name of the question.
200      */
201     public String getVariable() {
202         return element.attributeValue("var");
203     }
204 
205     /**
206      * Returns an Iterator for the default values of the question if the question is part
207      * of a form to fill out. Otherwise, returns an Iterator for the answered values of
208      * the question.
209      *
210      * @return an Iterator for the default values or answered values of the question.
211      */
212     @SuppressWarnings("unchecked")
213     public List<String> getValues() {
214         List<String> answer = new ArrayList<String>();
215         for (Iterator<Element> it = element.elementIterator("value"); it.hasNext();) {
216             answer.add(it.next().getTextTrim());
217         }
218         return answer;
219     }
220 
221 	/**
222 	 * Returns the first value from the FormField, or 'null' if no value has
223 	 * been set.
224 	 * 
225 	 * @param formField
226 	 *            The field from which to return the first value.
227 	 * @return String based value, or 'null' if the FormField has no values.
228 	 */
229     @SuppressWarnings("unchecked")
230 	public String getFirstValue()
231 	{
232         for (Iterator<Element> it = element.elementIterator("value"); it.hasNext();) {
233             return it.next().getTextTrim();
234         }
235         
236         return null;
237 	}

240      * Returns an indicative of the format for the data to answer. Valid formats are:
241      * 
242      * <ul>
243      * <li>text-single -> single line or word of text
244      * <li>text-private -> instead of showing the user what they typed, you show ***** to
245      * protect it
246      * <li>text-multi -> multiple lines of text entry
247      * <li>list-single -> given a list of choices, pick one
248      * <li>list-multi -> given a list of choices, pick one or more
249      * <li>boolean -> 0 or 1, true or false, yes or no. Default value is 0
250      * <li>fixed -> fixed for putting in text to show sections, or just advertise your web
251      * site in the middle of the form
252      * <li>hidden -> is not given to the user at all, but returned with the questionnaire
253      * <li>jid-single -> Jabber ID - choosing a JID from your roster, and entering one based
254      * on the rules for a JID.
255      * <li>jid-multi -> multiple entries for JIDs
256      * </ul>
257      *
258      * @return format for the data to answer.
259      */
260     public Type getType() {
261         String type = element.attributeValue("type");
262         if (type != null) {
263             return Type.fromXMPP(type);
264         }
265         return null;
266     }

269      * Returns the label of the question which should give enough information to the user to
270      * fill out the form.

272      * @return label of the question.
273      */
274     public String getLabel() {
275         return element.attributeValue("label");
276     }

278     /**
279      * Returns a description that provides extra clarification about the question. This information

281      * before the question.
282      * 
283      * If the question is of type FIXED then the description should remain empty.
284      *
285      * @return description that provides extra clarification about the question.

288         return element.elementTextTrim("desc");
289     }

331     /**
332      * Type-safe enumeration to represent the field type of the Data forms.
333      *
334      * Implementation note: XMPP error conditions use "-" characters in
335      * their names such as "jid-multi". Because "-" characters are not valid
336      * identifier parts in Java, they have been converted to "_" characters in
337      * the  enumeration names, such as <tt>jid_multi</tt>. The {@link #toXMPP()} and
338      * {@link #fromXMPP(String)} methods can be used to convert between the
339      * enumertation values and Type code strings.
340      */
341     public enum Type {
342         /**
343          * The field enables an entity to gather or provide an either-or choice between two
344          * options. The allowable values are 1 for yes/true/assent and 0 for no/false/decline.
345          * The default value is 0.
346          */
347         boolean_type("boolean"),
348 
349         /**
350          * The field is intended for data description (e.g., human-readable text such as
351          * "section" headers) rather than data gathering or provision. The <value/> child
352          * SHOULD NOT contain newlines (the \n and \r characters); instead an application
353          * SHOULD generate multiple fixed fields, each with one <value/> child.
354          */
355         fixed("fixed"),
356 
357         /**
358          * The field is not shown to the entity providing information, but instead is
359          * returned with the form.
360          */
361         hidden("hidden"),
362 
363         /**
364          * The field enables an entity to gather or provide multiple Jabber IDs.
365          */
366         jid_multi("jid-multi"),
367 
368         /**
369          * The field enables an entity to gather or provide multiple Jabber IDs.
370          */
371         jid_single("jid-single"),
372 
373         /**
374          * The field enables an entity to gather or provide one or more options from
375          * among many.
376          */
377         list_multi("list-multi"),
378 
379         /**
380          * The field enables an entity to gather or provide one option from among many.
381          */
382         list_single("list-single"),
383 
384         /**
385          * The field enables an entity to gather or provide multiple lines of text.
386          */
387         text_multi("text-multi"),
388 
389         /**
390          * The field enables an entity to gather or provide a single line or word of text,
391          * which shall be obscured in an interface (e.g., *****).
392          */
393         text_private("text-private"),
394 
395         /**
396          * The field enables an entity to gather or provide a single line or word of text,
397          * which may be shown in an interface. This field type is the default and MUST be
398          * assumed if an entity receives a field type it does not understand.
399          */
400         text_single("text-single");
401 
402         /**
403          * Converts a String value into its Type representation.
404          *
405          * @param type the String value.
406          * @return the type corresponding to the String.
407          */
408         public static Type fromXMPP(String type) {
409             if (type == null) {
410                 throw new NullPointerException();
411             }
412             type = type.toLowerCase();
413             if (boolean_type.toXMPP().equals(type)) {
414                 return boolean_type;
415             }
416             else if (fixed.toXMPP().equals(type)) {
417                 return fixed;
418             }
419             else if (hidden.toXMPP().equals(type)) {
420                 return hidden;
421             }
422             else if (jid_multi.toXMPP().equals(type)) {
423                 return jid_multi;
424             }
425             else if (jid_single.toXMPP().equals(type)) {
426                 return jid_single;
427             }
428             else if (list_multi.toXMPP().equals(type)) {
429                 return list_multi;
430             }
431             else if (list_single.toXMPP().equals(type)) {
432                 return list_single;
433             }
434             else if (text_multi.toXMPP().equals(type)) {
435                 return text_multi;
436             }
437             else if (text_private.toXMPP().equals(type)) {
438                 return text_private;
439             }
440             else if (text_single.toXMPP().equals(type)) {
441                 return text_single;
442             }
443             else {
444                 throw new IllegalArgumentException("Type invalid:" + type);
445             }
446         }
447 
448         private String value;
449 
450         private Type(String value) {
451             this.value = value;
452         }
453 
454         /**
455          * Returns the Field Type as a valid Field Type code string.
456          *
457          * @return the Field Type value.
458          */
459         public String toXMPP() {
460             return value;
461         }
