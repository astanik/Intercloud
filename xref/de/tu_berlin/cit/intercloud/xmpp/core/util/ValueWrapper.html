18  package de.tu_berlin.cit.intercloud.xmpp.core.util;
19  
20  import java.io.Serializable;
21  import java.util.Map;
22  
23  
24  /**
25   * A wrapper implementation for cached values, suitable for {@link Map} based
26   * caches where a significant portion of keys matches the corresponding value
27   * exactly.
28   * 
29   * @author Guus der Kinderen, guus.der.kinderen@gmail.com
30   */
31  public class ValueWrapper<V> implements Serializable {
32  
33  	private static final long serialVersionUID = -9054807517127343156L;
34  
35  	/**
36  	 * Indication of how the key of this cache entry represents the cache value.
37  	 */
38  	public enum Representation {
39  
40  		/**
41  		 * The key that maps to this {@link ValueWrapper} instance cannot be
42  		 * used to generate a valid value.
43  		 */
44  		ILLEGAL,
45  
46  		/**
47  		 * The generated value based on the key that maps to this
48  		 * {@link ValueWrapper} would be an exact duplicate of the key. To
49  		 * conserve memory, this wrapped value instance will not have a value
50  		 * set. Use the key that points to this wrapper instead.
51  		 */
52  		USE_KEY,
53  
54  		/**
55  		 * The key that maps to this {@link ValueWrapper} can be used to
56  		 * generate a valid value. The generated value is wrapped in this
57  		 * {@link ValueWrapper} instance.
58  		 */
59  		USE_VALUE
60  	};
61  
62  	/**
63  	 * The value that is wrapped.
64  	 */
65  	private final V value;
66  
67  	/**
68  	 * Indicates how the key that maps to this value can be used to extract the
69  	 * value from the cache entry.
70  	 */
71  	private final Representation representation;
72  
73  	/**
74  	 * Constructs an empty wrapper. This wrapper either is used to indicate that
75  	 * the key that maps to this value: cannot be used to generate a valid
76  	 * value, or, is an exact duplicate of the generated value.
77  	 * 
78  	 * An {@link IllegalArgumentException} is thrown if the argument is
79  	 * <tt>USE_VALUE</tt>.
80  	 * 
81  	 * @param representation
82  	 *            Key representation indicator.
83  	 */
84  	public ValueWrapper(Representation representation) {
85  		if (representation == Representation.USE_VALUE) {
86  			throw new IllegalArgumentException();
87  		}
88  
89  		this.representation = representation;
90  		this.value = null;
91  	}
92  
93  	/**
94  	 * Wraps a value while using the <tt>USE_VALUE</tt> representation.
95  	 * 
96  	 * @param value
97  	 *            The value that is wrapped.
98  	 */
99  	public ValueWrapper(V value) {
100 		this.representation = Representation.USE_VALUE;
101 		this.value = value;
102 	}
103 
104 	/**
105 	 * Returns the wrapped value, or <tt>null</tt> if the representation used in
106 	 * this instance is not USE_VALUE;
107 	 * 
108 	 * @return the wrapped value.
109 	 */
110 	public V getValue() {
111 		return value;
112 	}
113 
114 	public Representation getRepresentation() {
115 		return representation;
116 	}
117 }
