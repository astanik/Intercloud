30  /**
31   * IQ (Info/Query) packet. IQ packets are used to get and set information
32   * on the server, including authentication, roster operations, and creating
33   * accounts. Each IQ packet has a specific type that indicates what type of action
34   * is being taken: "get", "set", "result", or "error".<p>
35   *
36   * IQ packets can contain a single child element that exists in a extended XML
37   * namespace.
38   */
39  public class IQ extends Packet {

43      // Sequence and random number generator used for creating unique ID's.
44      private static int sequence = 0;
45      private static Random random = new Random();

47      /**
48       * Constructs a new IQ with an automatically generated ID and a type
49       * of {@link Type#get IQ.Type.get}.
50       */
51      public IQ() {
52          this.element = docFactory.createDocument().addElement("iq");
53          String id = String.valueOf(random.nextInt(1000) + "-" + sequence++);
54          setType(Type.get);
55          setID(id);
56      }

59       * Constructs a new IQ using the specified type. A packet ID will

65          this.element = docFactory.createDocument().addElement("iq");
66          setType(type);
67          String id = String.valueOf(random.nextInt(1000) + "-" + sequence++);
68          setID(id);
69      }

167     /**
168      * Returns the child element of this IQ. IQ packets may have a single child
169      * element in an extended namespace. This is a convenience method to
170      * avoid manipulating the underlying packet's Element instance directly.<p>
171      *
172      * An IQ child element in extended namespaces is used to extend the features
173      * of XMPP. Although any valid XML can be included in a child element
174      * in an extended namespace, many common features have been standardized
175      * as <a href="http://xmpp.org/extensions/">XMPP Extension Protocols</a>
176      * (XEPs).
177      *
178      * @return the child element.
179      */
180     @SuppressWarnings("unchecked")
181     public Element getChildElement() {
182         List<Element> elements = element.elements();
183         if (elements.isEmpty()) {
184             return null;
185         }
186         else {
187             // Search for a child element that is in a different namespace.
188             for (int i=0; i<elements.size(); i++) {
189                 Element element = elements.get(i);
190                 String namespace = element.getNamespaceURI();
191                 if (!namespace.equals("") && !namespace.equals("jabber:client") &&
192                         !namespace.equals("jabber:server"))
193                 {
194                     return element;
195                 }
196             }
197             return null;
198         }
199     }

201     /**
202      * Sets the child element of this IQ. IQ packets may have a single child
203      * element in an extended namespace. This is a convenience method to
204      * avoid manipulating this underlying packet's Element instance directly.<p>
205      *
206      * A sample use of this method might look like the following:
207      * <pre>
208      * IQ iq = new IQ("time_1");
209      * iq.setTo("mary@example.com");
210      * iq.setType(IQ.Type.GET);
211      * iq.setChildElement(docFactory.createElement("query", "jabber:iq:time"));</pre><p>
212      *
213      * An IQ child element in extended namespaces is used to extend the features
214      * of XMPP. Although any valid XML can be included in a child element
215      * in an extended namespace, many common features have been standardized
216      * as <a href="http://xmpp.org/extensions/">XMPP Extension Protocols</a>
217      * (XEPs).
218      *
219      * @param childElement the child element.
220      */
221     @SuppressWarnings("unchecked")
222     public void setChildElement(Element childElement) {
223         for (Iterator<Element> i=element.elementIterator(); i.hasNext(); ) {
224             element.remove(i.next());
225         }
226         element.add(childElement);
227     }
228 
229     /**
230      * Sets the child element of this IQ by constructing a new Element with the
231      * given name and namespace. The newly created child element is returned.
232      * IQ packets may have a single child element in an extended namespace.
233      * This method is a convenience method to avoid manipulating the underlying
234      * packet's Element instance directly.<p>
235      *
236      * In some cases, additional custom sub-elements must be added to an IQ child
237      * element (called packet extensions). For example, when adding a data form to
238      * an IQ response. See {@link #addExtension(PacketExtension)}.<p>
239      *
240      * A sample use of this method might look like the following:
241      * <pre>
242      * IQ iq = new IQ("time_1");
243      * iq.setTo("mary@example.com");
244      * iq.setType(IQ.Type.GET);
245      * iq.setChildElement("query", "jabber:iq:time");</pre>
246      *
247      * @param name the child element name.
248      * @param namespace the child element namespace.
249      * @return the newly created child element.
250      */
251     @SuppressWarnings("unchecked")
252     public Element setChildElement(String name, String namespace) {
253         for (Iterator<Element> i=element.elementIterator(); i.hasNext(); ) {
254             element.remove(i.next());
255         }
256         return element.addElement(name, namespace);
257     }

259     /**
260      * Adds the element contained in the PacketExtension to the child element of the IQ
261      * packet. IQ packets, unlike the other packet types, have a unique child element that
262      * holds the packet extensions. If an extension is added to an IQ packet that does
263      * not have a child element then an IllegalStateException will be thrown.<p>
264      *
265      * It is important that this is the first and last time the element contained in
266      * PacketExtension is added to another Packet. Otherwise, a runtime error will be
267      * thrown when trying to add the PacketExtension's element to the Packet's element.
268      * Future modifications to the PacketExtension will be reflected in this Packet.<p>
269      *
270      * Note: packet extensions on IQ packets are only for use in specialized situations.
271      * In most cases, you should only need to set the child element of the IQ.
272      *
273      * @param extension the PacketExtension whose element will be added to this Packet's element.
274      */
275     public void addExtension(PacketExtension extension) {
276         Element childElement = getChildElement();
277         if (childElement == null) {
278             throw new IllegalStateException("Cannot add packet extension when child element is null");
279         }
280         // Add the extension to the child element
281         childElement.add(extension.getElement());
282     }

284     /**
285      * Returns a {@link PacketExtension} on the first element found in this packet's
286      * child element for the specified <tt>name</tt> and <tt>namespace</tt> or <tt>null</tt> if
287      * none was found. If the IQ packet does not have a child element then <tt>null</tt>
288      * will be returned.<p>
289      *
290      * Note: packet extensions on IQ packets are only for use in specialized situations.
291      * In most cases, you should only need to set the child element of the IQ.
292      *
293      * @param name the child element name.
294      * @param namespace the child element namespace.
295      * @return a PacketExtension on the first element found in this packet for the specified
296      *         name and namespace or <tt>null</tt> if none was found.
297      */
298     @SuppressWarnings("unchecked")
299     public PacketExtension getExtension(String name, String namespace) {
300         Element childElement = getChildElement();
301         if (childElement == null) {
302             return null;
303         }
304         // Search for extensions in the child element
305         List<Element> extensions = childElement.elements(QName.get(name, namespace));
306         if (!extensions.isEmpty()) {
307             Class<? extends PacketExtension> extensionClass = PacketExtension.getExtensionClass(name, namespace);
308             if (extensionClass != null) {
309                 try {
310                     Constructor<? extends PacketExtension> constructor = extensionClass.getDeclaredConstructor(new Class[]{
311                         Element.class});
312                     return constructor.newInstance(new Object[]{
313                         extensions.get(0)});
314                 }
315                 catch (Exception e) {
316                     Log.warn("Packet extension (name "+name+", namespace "+namespace+") cannot be found.", e);
317                 }
318             }
319         }
320         return null;
321     }

323     /**
324      * Deletes the first element whose element name and namespace matches the specified
325      * element name and namespace in this packet's child element. If the
326      * IQ packet does not have a child element then this method does nothing and returns
327      * <tt>false</tt>.<p>
328      *
329      * Notice that this method may remove any child element that matches the specified
330      * element name and namespace even if that element was not added to the Packet using a
331      * {@link PacketExtension}.<p>
332      *
333      * Note: packet extensions on IQ packets are only for use in specialized situations.
334      * In most cases, you should only need to set the child element of the IQ.
335      *
336      * @param name the child element name.
337      * @param namespace the child element namespace.
338      * @return true if a child element was removed.
339      */
340     @SuppressWarnings("unchecked")
341 	public boolean deleteExtension(String name, String namespace) {
342         Element childElement = getChildElement();
343         if (childElement == null) {
344             return false;
345         }
346         // Delete extensions in the child element
347         List<Element> extensions = childElement.elements(QName.get(name, namespace));
348         if (!extensions.isEmpty()) {
349             childElement.remove(extensions.get(0));
350             return true;
351         }
352         return false;
353     }

364     /**
365      * Convenience method to create a new {@link Type#result IQ.Type.result} IQ based
366      * on a {@link Type#get IQ.Type.get} or {@link Type#set IQ.Type.set} IQ. The new
367      * packet will be initialized with:<ul>
368      *
369      *      <li>The sender set to the recipient of the originating IQ.
370      *      <li>The recipient set to the sender of the originating IQ.
371      *      <li>The type set to {@link Type#result IQ.Type.result}.
372      *      <li>The id set to the id of the originating IQ.
373      * </ul>
374      *
375      * @param iq the {@link Type#get IQ.Type.get} or {@link Type#set IQ.Type.set} IQ packet.
376      * @throws IllegalArgumentException if the IQ packet does not have a type of
377      *      {@link Type#get IQ.Type.get} or {@link Type#set IQ.Type.set}.
378      * @return a new {@link Type#result IQ.Type.result} IQ based on the originating IQ.
379      */
380     public static IQ createResultIQ(IQ iq) {
381         if (!(iq.getType() == Type.get || iq.getType() == Type.set)) {
382             throw new IllegalArgumentException(
383                     "IQ must be of type 'set' or 'get'. Original IQ: " + iq.toXML());
384         }
385         IQ result = new IQ(Type.result, iq.getID());
386         result.setFrom(iq.getTo());
387         result.setTo(iq.getFrom());
388         return result;
389     }

391     /**
392      * Type-safe enumeration to represent the type of the IQ packet. The types are:
393      *
394      * <ul>
395      *      <li>IQ.Type.get -- the IQ is a request for information or requirements.
396      *      <li>IQ.Type.set -- the IQ provides required data, sets new values, or
397      *          replaces existing values.
398      *      <li>IQ.Type.result -- the IQ is a response to a successful get or set request.
399      *      <li>IQ.Type.error -- an error has occurred regarding processing or delivery of a
400      *          previously-sent get or set.
401      * </ul>
402      *
403      * If {@link #get IQ.Type.get} or {@link #set IQ.Type.set} is received the response
404      * must be {@link #result IQ.Type.result} or {@link #error IQ.Type.error}. The id of the
405      * originating {@link #get IQ.Type.get} of {@link #set IQ.Type.set} IQ must be preserved
406      * when sending {@link #result IQ.Type.result} or {@link #error IQ.Type.error}.
407      */
408     public enum Type {
409 
410         /**
411          * The IQ is a request for information or requirements.
412          */
413         get,
414 
415         /**
416          * The IQ provides required data, sets new values, or
417          * replaces existing values.
418          */
419         set,
420 
421         /**
422          * The IQ is a response to a successful get or set request.
423          */
424         result,
425 
426         /**
427          * An error has occurred regarding processing or delivery of a
428          * previously-sent get or set.
429          */
430         error;
431 
432     }
