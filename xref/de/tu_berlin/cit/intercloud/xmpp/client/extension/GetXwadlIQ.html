17  package de.tu_berlin.cit.intercloud.xmpp.client.extension;
18  
19  
20  import org.jivesoftware.smack.packet.IQ;
21  import org.jivesoftware.smack.packet.id.StanzaIdUtil;
22  
23  import de.tu_berlin.cit.intercloud.xmpp.rest.XmppURI;
24  
25  public class GetXwadlIQ extends IQ {
26  
27  	public static final String ELEMENT = "resource_type";
28  	public static final String NAMESPACE = "urn:xmpp:rest-xwadl";
29  
30  	final private XmppURI uri;
31  	
32  	public GetXwadlIQ(XmppURI uri) {
33  		super(ELEMENT, NAMESPACE);
34  		this.setStanzaId(StanzaIdUtil.newStanzaId());
35  		this.uri = uri;
36  		this.setType(Type.get);
37  		this.setTo(uri.getJID());
38  	}
39  
40  	@Override
41  	protected IQChildElementXmlStringBuilder getIQChildElementBuilder(
42  			IQChildElementXmlStringBuilder xml) {
43  		// set resource path
44  		xml.attribute("path", this.uri.getPath());
45  //		xml.rightAngleBracket();
46  		// mark that no further elements are present
47  		xml.setEmptyElement();
48  		return xml;
49  	}
50  
51  }
