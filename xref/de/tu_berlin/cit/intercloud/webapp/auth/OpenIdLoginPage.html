18  
19  import org.apache.wicket.markup.html.form.Button;
20  import org.apache.wicket.markup.html.form.Form;
21  import org.apache.wicket.markup.html.form.TextField;
22  import org.apache.wicket.model.CompoundPropertyModel;
23  import org.apache.wicket.model.Model;
24  import org.apache.wicket.request.http.handler.RedirectRequestHandler;
25  import org.openid4java.discovery.DiscoveryInformation;
26  import org.openid4java.message.AuthRequest;
27  
28  import de.tu_berlin.cit.intercloud.webapp.IntercloudWebSession;
29  
30  /**
31   * Page that allows users to login into the website with their Open Id account.
32   * 
33   * @author Alexander Stanik
34   */
35  @SuppressWarnings("serial")
36  public class OpenIdLoginPage extends AbstractAuthPage {
37  
38  	/**
39  	 * Constructor
40  	 */
41  	public OpenIdLoginPage() {
42  		super();
43  
44  		// add form
45  		add(new OpenIdRegistrationForm(this));
46  	}
47  
48  	/**
49  	 * The Open ID login form used for this Page.
50  	 * 
51  	 */
52  	public class OpenIdRegistrationForm extends Form<OpenIdUser> {
53  
54  		public OpenIdRegistrationForm(final OpenIdLoginPage loginPage) {
55  
56  			super("form", new CompoundPropertyModel<OpenIdUser>(
57  					new OpenIdUser()));
58  			//
59  
60  			final OpenIdUser formModel = (OpenIdUser) this
61  					.getDefaultModelObject();
62  
63  			// setModelObject(formModel);
64  			//
65  			add(new TextField<String>("openId").setRequired(true).setLabel(
66  					new Model<String>("Your Open ID")));
67  			// This is the "business end" of making the authentication request.
68  			Button confirmOpenIdButton = new Button("confirmOpenIdButton") {
69  				@Override
70  				public void onSubmit() {
71  					// Delegate to Open ID code
72  					DiscoveryInformation discoveryInformation = OpenIdService
73  							.performDiscoveryOnUserSuppliedIdentifier(formModel
74  									.getOpenId());
75  					// Store the discovery results in session.
76  					IntercloudWebSession session = (IntercloudWebSession) this
77  							.getSession();
78  					session.setDiscoveryInformation(discoveryInformation);
79  					// Create the AuthRequest
80  					AuthRequest authRequest = OpenIdService
81  							.createOpenIdAuthRequest(discoveryInformation,
82  									OpenIdService.getReturnToUrl());
83  					// Now take the AuthRequest and forward it on to the OP
84  					RedirectRequestHandler handler = new RedirectRequestHandler(
85  							authRequest.getDestinationUrl(true));
86  					this.getRequestCycle().scheduleRequestHandlerAfterCurrent(
87  							handler);
88  				}
89  			};
90  			add(confirmOpenIdButton);
91  		}
92  	}
