17  package de.tu_berlin.cit.intercloud.webapp.auth;
18  
19  import org.apache.wicket.RestartResponseAtInterceptPageException;
20  import org.apache.wicket.request.mapper.parameter.PageParameters;
21  import org.openid4java.discovery.DiscoveryInformation;
22  
23  import de.tu_berlin.cit.intercloud.webapp.IntercloudWebSession;
24  
25  /**
26   * This class represents the a registration page, which receives the
27   * authentication response from the OpenID Provider (OP) and verifies the
28   * response with openid4java. It also provides a way to save the information
29   * retrieved from the OP in a database and the user in the session.
30   * 
31   */
32  @SuppressWarnings("serial")
33  public class UserRegistrationPage extends AbstractAuthPage {
34  
35  	/**
36  	 * Constructor
37  	 */
38  	public UserRegistrationPage() {
39  		this(new PageParameters());
40  	}
41  
42  	/**
43  	 * Constructor called by Wicket with an authentication response.
44  	 * 
45  	 * @param pageParameters
46  	 *            The request parameters that are the response parameters from
47  	 *            the OP.
48  	 */
49  	public UserRegistrationPage(PageParameters pageParameters) {
50  		super();
51  		IntercloudWebSession session = (IntercloudWebSession) getSession();
52  		OpenIdUser user = null;
53  
54  		if (!pageParameters.isEmpty()) {
55  			// If this is a return trip (the OP will redirect here once
56  			// authentication
57  			// is compelete), then verify the response.
58  			if (pageParameters.get("is_return").toBoolean()) {
59  				//
60  				// Grab the session object so we can let openid4java do
61  				// verification.
62  				//
63  				DiscoveryInformation discoveryInformation = session
64  						.getDiscoveryInformation();
65  				//
66  				// Delegate to the Service object to do verification. It will
67  				// return
68  				// / the RegistrationModel to use to display the information
69  				// that was
70  				// / retrieved from the OP about the User-Supplied identifier.
71  				// The
72  				// / RegistrationModel reference will be null if there was a
73  				// problem
74  				// / (check the logs for more information if this happens).
75  				//
76  				user = OpenIdService.processReturn(discoveryInformation,
77  						pageParameters, OpenIdService.getReturnToUrl());
78  				if (user == null) {
79  					//
80  					// Oops, something went wrong. Display a message on the
81  					// screen.
82  					// / Check the logs for more information.
83  					//
84  					error("Open ID Confirmation Failed. No information was retrieved from the OpenID Provider.");
85  				}
86  			} else {
87  				error("Corrupt page call. No information was retrieved from the OpenID Provider.");
88  			}
89  		} else {
90  			error("Open ID Confirmation Failed. No page parameters were retrieved from the OpenID Provider.");
91  		}
92  
93  		if (user.getFullName().isEmpty() || user.getEmailAddress().isEmpty()) {
94  			error("Open ID Confirmation Failed. Required information were not retrieved from the OpenID Provider.");
95  		} 
96  		
97  		// Redirect to the webapp
98  		if(session.authenticate(user))
99  			throw new RestartResponseAtInterceptPageException(getApplication().getHomePage());
100 //		throw new ImmediateRedirectException(getApplication().getHomePage());
101 	}
102 
103 }
