57  	/**
58  	 * Perform discovery on the User-Supplied identifier and return the
59  	 * DiscoveryInformation object that results from Association with the
60  	 * OP.
61  	 * 
62  	 * @param userSuppliedIdentifier The User-Supplied identifier. It may already
63  	 *  be normalized.
64  	 *
65  	 * @return DiscoveryInformation The resulting DisoveryInformation object
66  	 *  returned by openid4java following successful association with the OP.
67  	 */
68  	@SuppressWarnings("unchecked")
69  	public static DiscoveryInformation performDiscoveryOnUserSuppliedIdentifier(String userSuppliedIdentifier) {
70  		ConsumerManager consumerManager = getConsumerManager();
71  		try {
72  			// Perform discover on the User-Supplied Identifier
73  			List<DiscoveryInformation> discoveries = consumerManager.discover(userSuppliedIdentifier);
74  			// Pass the discoveries to the associate() method...
75  			return consumerManager.associate(discoveries);
76  
77  		} catch (DiscoveryException e) {
78  			String message = "Error occurred during discovery!";
79  			log.error(message, e);
80  			throw new RuntimeException(message, e);
81  		}
82  	}
83  	
84  	/**
85  	 * Create an OpenID Auth Request, using the DiscoveryInformation object
86  	 * return by the openid4java library.
87  	 * 
88  	 * This method uses not only the Simple Registration Extension but also 
89  	 * the Attribute Exchange Extension to grant the Relying Party (RP).
90  	 * 
91  	 * @param discoveryInformation The DiscoveryInformation that should have
92  	 *  been previously obtained from a call to performDiscoveryOnUserSuppliedIdentifier().
93  	 *  
94  	 * @param returnToUrl The URL to which the OP will redirect once the
95  	 *   authentication call is complete.
96  	 *  
97  	 * @return AuthRequest A "good-to-go" AuthRequest object packed with all
98  	 *  kinds of great OpenID goodies for the OpenID Provider (OP). The caller
99  	 *  must take this object and forward it on to the OP.
100 	 */
101 	public static AuthRequest createOpenIdAuthRequest(DiscoveryInformation discoveryInformation, String returnToUrl) {
102 		try {
103 			// Create the AuthRequest object
104 			AuthRequest ret = getConsumerManager().authenticate(discoveryInformation, returnToUrl);
105 			// Create the Simple Registration Request
106 			ret.addExtension(createSimpleRegistrationExtension());
107 			// Create the Attribute Exchange Request
108 			ret.addExtension(createAttributeExchangeExtension());
109 			
110 			return ret;
111 			
112 		} catch (Exception e) {
113 			String message = "Exception occurred while building AuthRequest object!";
114 			log.error(message, e);
115 			throw new RuntimeException(message, e);
116 		}
117 	}

119 	/**
120 	 * Returns a Simple Registration Extension for full name and email address.
121 	 * 
122 	 */
123 	private static SRegRequest createSimpleRegistrationExtension() {
124 		SRegRequest sRegRequest = SRegRequest.createFetchRequest();
125 		sRegRequest.addAttribute("fullname", true);
126 		sRegRequest.addAttribute("email", true);
127 		return sRegRequest;
128 	}
129 	
130 	/**
131 	 * Returns a Attribute Exchange Extension for first name, last name and one email address.
132 	 * 
133 	 */
134 	private static FetchRequest createAttributeExchangeExtension() throws MessageException {
135 		FetchRequest fetch = FetchRequest.createFetchRequest();
136 		fetch.addAttribute("FirstName", "http://schema.openid.net/namePerson/first", true);
137 		fetch.addAttribute("LastName", "http://schema.openid.net/namePerson/last", true);
138 		fetch.addAttribute("Email", "http://schema.openid.net/contact/email", true);
139 		// wants up to one email address
140 		fetch.setCount("Email", 1);
141 		return fetch;
142 	}

144 	/**
145 	 * Processes the returned information from an authentication request
146 	 * from the OP.
147 	 * 
148 	 * @param discoveryInformation DiscoveryInformation that was created earlier
149 	 *  in the conversation (by openid4java). This will need to be verified with
150 	 *  openid4java to make sure everything went smoothly and there are no
151 	 *  possible problems.
152 	 *  
153 	 * @param pageParameters PageParameters passed to the page handling the
154 	 *  return verification.
155 	 *  
156 	 * @param returnToUrl The "return to" URL that was passed to the OP. It must
157 	 *  match exactly, or openid4java will issue a verification failed message
158 	 *  in the logs.
159 	 *  
160 	 * @return OpenIdUser - An empty object if there was a problem, or an
161 	 *  object with parameters filled in as completely as possible from the
162 	 *  information available from the OP.
163 	 */
164 	public static OpenIdUser processReturn(DiscoveryInformation discoveryInformation, PageParameters pageParameters, String returnToUrl) {
165 		OpenIdUser ret = new OpenIdUser();
166 		ParameterList response = convertWicketPageParameters(pageParameters);
167 		// Verify the Information returned from the OP
168 		try {
169 			VerificationResult verificationResult = getConsumerManager().verify(returnToUrl, response, discoveryInformation);
170 			Identifier verifiedIdentifier = verificationResult.getVerifiedId();
171 			if (verifiedIdentifier != null) {
172 				ret.setOpenId(verifiedIdentifier.getIdentifier());
173 				AuthSuccess authSuccess = (AuthSuccess)verificationResult.getAuthResponse();
174 				ret = getValuesFromMessageExtension(ret, authSuccess);
175 			}
176 		} catch (Exception e) {
177 			String message = "Exception occurred while verifying response!";
178 			log.error(message, e);
179 			throw new RuntimeException(message, e);
180 		}
181 		return ret;
182 	}

184 	private static ParameterList convertWicketPageParameters(final PageParameters pageParameters) {
185 		// create parameter list
186 		ParameterList response = new ParameterList();
187 		// get key set
188 		Set<String> keys = pageParameters.getNamedKeys();
189 		Iterator<String> iter = keys.iterator();
190 		// copy all keys and values
191 		while (iter.hasNext()) {
192 			String theKey = iter.next();
193 			String theValue = pageParameters.get(theKey).toString();
194 			response.set(new Parameter(theKey, theValue));
195 		}
196 		// Log parameter list
197 		log.info(pageParameters.toString());
198 		return response;
199 	}

201 	private static OpenIdUser getValuesFromMessageExtension(OpenIdUser user, final AuthSuccess auth) throws MessageException {
202 		// for Simple Registration Extension
203 		if (auth.hasExtension(SRegMessage.OPENID_NS_SREG)) {
204 			MessageExtension extension = auth.getExtension(SRegMessage.OPENID_NS_SREG);
205 			if (extension instanceof SRegResponse) {
206 				SRegResponse sRegResponse = (SRegResponse)extension;
207 				String value = sRegResponse.getAttributeValue("email");
208 				if (value != null) {
209 				  user.setEmailAddress(value);
210 				}
211 				value = sRegResponse.getAttributeValue("fullname");
212 				if (value != null) {
213 					user.setFullName(value);
214 				}
215 			}
216 		} 
217 		// for Attribute Exchange Extension
218 		else if(auth.hasExtension(AxMessage.OPENID_NS_AX)) {
219 			MessageExtension extension = auth.getExtension(AxMessage.OPENID_NS_AX);
220 
221 		    if (extension instanceof FetchResponse)
222 		    {
223 		        FetchResponse fetchResp = (FetchResponse)extension;
224 		        
225 		        String firstName = fetchResp.getAttributeValue("FirstName");
226 		        String lastName = fetchResp.getAttributeValue("LastName");
227 				if (firstName != null && lastName != null) {
228 					user.setFullName(firstName + " " + lastName);
229 				}
230 		        
231 		        // can have multiple values
232 		        List<?> emails = fetchResp.getAttributeValues("Email");
233 		        if(emails != null) {
234 		        	if(!emails.isEmpty())
235 		        		user.setEmailAddress(emails.get(0).toString());
236 		        }
237 		    }
238 
239 		}
240 		return user;
241 	}

243 	private static ConsumerManager consumerManager;
244 	/**
245 	 * Retrieves an instance of the ConsumerManager object. It is static
246 	 * (see note in Class-level JavaDoc comments above) because openid4java
247 	 * likes it that way.
248 	 * 
249 	 * @return ConsumerManager - The ConsumerManager object that handles
250 	 *  communication with the openid4java API.
251 	 */
252 	private static ConsumerManager getConsumerManager() {
253 		if (consumerManager == null) {
254 			consumerManager = new ConsumerManager();
255 			consumerManager.setAssociations(new InMemoryConsumerAssociationStore());
256 			consumerManager.setNonceVerifier(new InMemoryNonceVerifier(10000));
257 		}
258 		return consumerManager;
259 	}
