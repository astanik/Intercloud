35  /**
36   * Frame used as page navigation panel.
37   * 
38   * @author Alexander Stanik
39   */
40  @SuppressWarnings("serial")
41  public class NavigationFrame extends WebPage {
42  
43  	/**
44  	 * Link that, when clicked, changes the frame target's frame class (and as
45  	 * that is a shared model which is also being used by the 'master page'
46  	 * {@link BodyFrame}, changes are immediately reflected) and set the
47  	 * response page to the top level page {@link BodyFrame}. Tags that use this
48  	 * link should have a <code>target="_parent"</code> attribute, so that the
49  	 * top frame will be refreshed.
50  	 */
51  	private static class ChangeFramePageLink extends Link<NavigationItem> {
52  		private static final long serialVersionUID = 1L;
53  
54  		/** parent frame class. */
55  		private final BodyFrame bodyFrame;
56  
57  		/** this link's target. */
58  		private final Class<? extends BasePage> pageClass;
59  
60  		/**
61  		 * Construct.
62  		 * 
63  		 * @param <C>
64  		 * 
65  		 * @param id
66  		 * @param bodyFrame
67  		 * @param pageClass
68  		 */
69  		public <C extends BasePage> ChangeFramePageLink(String id, BodyFrame bodyFrame, Class<C> pageClass) {
70  			super(id);
71  			this.bodyFrame = bodyFrame;
72  			this.pageClass = pageClass;
73  		}
74  
75  		/**
76  		 * @see org.apache.wicket.markup.html.link.Link#onClick()
77  		 */
78  		@Override
79  		public void onClick() {
80  			// change frame class
81  			bodyFrame.getFrameTarget().setFrameClass(pageClass);
82  
83  			// trigger re-rendering of the page
84  			setResponsePage(bodyFrame);
85  		}
86  	}
87  
88  	/**
89  	 * A panel that is only visible for users with role ADMIN.
90  	 */
91  	@AuthorizeAction(action = Action.RENDER, roles = Roles.ADMIN)
92  	private static class ForAdmins extends Panel {
93  		/**
94  		 * Construct.
95  		 * 
96  		 * @param id
97  		 */
98  		public ForAdmins(String id, final BodyFrame index) {
99  			super(id);
100 			
101 			// generate data-view to populate the table
102 			DataView<NavigationItem> naviView = new DataView<NavigationItem>("adminNavis",
103 					new ListDataProvider<NavigationItem>(index.getAdminNavigationList())) {
104 
105 				@Override
106 				protected void populateItem(Item<NavigationItem> item) {
107 					NavigationItem naviItem = item.getModelObject();
108 
109 					// link to the page
110 					ChangeFramePageLink link = new ChangeFramePageLink("linkToAdminPage", index, naviItem.getPageClass());
111 					link.add(new Label("adminLinkLabel", naviItem.getPageName()));
112 					item.add(link);
113 				}
114 			};
115 			
116 			add(naviView);
117 		}
118 	}
119 
120 	/**
121 	 * A panel that is only visible for users with role ADMIN or USER.
122 	 */
123 	@AuthorizeAction(action = Action.RENDER, roles = { Roles.ADMIN, Roles.USER })
124 	private static final class ForAdminsAndUsers extends Panel {
125 		/**
126 		 * Construct.
127 		 * 
128 		 * @param id
129 		 */
130 		public ForAdminsAndUsers(String id, final BodyFrame index) {
131 			super(id);
132 			BookmarkablePageLink<?> link = new BookmarkablePageLink<Void>("linkToLogout", LogoutPage.class);
133 			add(link);
134 
135 			// generate data-view to populate the table
136 			DataView<NavigationItem> naviView = new DataView<NavigationItem>("userNavis",
137 					new ListDataProvider<NavigationItem>(index.getUserNavigationList())) {
138 
139 				@Override
140 				protected void populateItem(Item<NavigationItem> item) {
141 					NavigationItem naviItem = item.getModelObject();
142 
143 					// link to the page
144 					ChangeFramePageLink link = new ChangeFramePageLink("linkToUserPage", index, naviItem.getPageClass());
145 					link.add(new Label("userLinkLabel", naviItem.getPageName()));
146 					item.add(link);
147 				}
148 			};
149 			
150 			add(naviView);
151 		}
152 	}
153 
154 	/**
155 	 * Constructor
156 	 * 
157 	 * @param index
158 	 *            parent frame class
159 	 */
160 	public NavigationFrame(BodyFrame index) {
161 		add(new ForAdminsAndUsers("forAdminsAndUsersPanel", index));
162 		add(new ForAdmins("forAdminsPanel", index));
163 	}
164 
165 	protected void refresh() {
166 		this.renderPage();
167 	}
168 
169 	/**
170 	 * No need for versioning this frame.
171 	 * 
172 	 * @see org.apache.wicket.Component#isVersioned()
173 	 */
174 	@Override
175 	public boolean isVersioned() {
176 		return false;
177 	}
178 }
