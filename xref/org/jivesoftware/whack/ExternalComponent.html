54  /**
55   * ExternalComponents are responsible for connecting and authenticating with a remote server and
56   * for sending and processing received packets. In fact, an ExternalComponent is a wrapper on a
57   * Component that provides remote connection capabilities. The actual processing of the packets is
58   * done by the wrapped Component.

125     public ExternalComponent(Component component, ExternalComponentManager manager, int maxThreads) {
126         this.component = component;
127         this.manager = manager;
128 
129         // Create a pool of threads that will process requests received by this component. If more
130         // threads are required then the command will be executed on the SocketReadThread process
131         threadPool = new ThreadPoolExecutor(maxThreads, maxThreads, 15, TimeUnit.SECONDS,
132                         new LinkedBlockingQueue<Runnable>(), new ThreadPoolExecutor.CallerRunsPolicy());
133     }

135     /**
136      * Generates a connection with the server and tries to authenticate. If an error occurs in any
137      * of the steps then a ComponentException is thrown.
138      *
139      * @param host          the host to connect with.
140      * @param port          the port to use.
141      * @param subdomain     the subdomain that this component will be handling.
142      * @throws ComponentException if an error happens during the connection and authentication steps.
143      */
144     public void connect(String host, int port, String subdomain) throws ComponentException {
145         try {
146             // Open a socket to the server
147             this.socket = new Socket();
148             socket.connect(new InetSocketAddress(host, port), manager.getConnectTimeout());
149             if (manager.getServerName() != null) {
150                 this.domain = subdomain + "." + manager.getServerName();
151             }
152             else {
153                 this.domain = subdomain;
154             }
155             this.subdomain = subdomain;
156             // Keep these variables that will be used in case a reconnection is required
157             this.host= host;
158             this.port = port;
159 
160             try {
161                 factory = XmlPullParserFactory.newInstance();
162                 reader = new XPPPacketReader();
163                 reader.setXPPFactory(factory);
164 
165                 reader.getXPPParser().setInput(new InputStreamReader(socket.getInputStream(),
166                         CHARSET));
167 
168                 // Get a writer for sending the open stream tag
169                 writer =
170                         new BufferedWriter(new OutputStreamWriter(socket.getOutputStream(),
171                                 CHARSET));
172                 // Open the stream.
173                 StringBuilder stream = new StringBuilder();
174                 stream.append("<stream:stream");
175                 stream.append(" xmlns=\"jabber:component:accept\"");
176                 stream.append(" xmlns:stream=\"http://etherx.jabber.org/streams\"");
177                 if (manager.isMultipleAllowed(subdomain)) {
178                     stream.append(" allowMultiple=\"true\"");
179                 }
180                 stream.append(" to=\"").append(domain).append("\">");
181                 writer.write(stream.toString());
182                 writer.flush();
183                 stream = null;
184 
185                 // Get the answer from the server
186                 XmlPullParser xpp = reader.getXPPParser();
187                 for (int eventType = xpp.getEventType(); eventType != XmlPullParser.START_TAG;) {
188                     eventType = xpp.next();
189                 }
190 
191                 // Set the streamID returned from the server
192                 connectionID = xpp.getAttributeValue("", "id");
193                 if (xpp.getAttributeValue("", "from") != null) {
194                     this.domain = xpp.getAttributeValue("", "from");
195                 }
196                 xmlSerializer = new XMLWriter(writer);
197 
198                 // Handshake with the server
199                 stream = new StringBuilder();
200                 stream.append("<handshake>");
201                 stream.append(StringUtils.hash(connectionID + manager.getSecretKey(subdomain)));
202                 stream.append("</handshake>");
203                 writer.write(stream.toString());
204                 writer.flush();
205                 stream = null;
206 
207                 // Get the answer from the server
208                 try {
209                     Element doc = reader.parseDocument().getRootElement();
210                     if ("error".equals(doc.getName())) {
211                         StreamError error = new StreamError(doc);
212                         // Close the connection
213                         socket.close();
214                         socket = null;
215                         // throw the exception with the wrapped error
216                         throw new ComponentException(error);
217                     }
218                     // Everything went fine
219                     // Start keep alive thread to send every 30 seconds of inactivity a heart beat
220                     keepAliveTask = new KeepAliveTask();
221                     TaskEngine.getInstance().scheduleAtFixedRate(keepAliveTask, 15000, 30000);
222 
223                     timeoutTask = new TimeoutTask();
224                     TaskEngine.getInstance().scheduleAtFixedRate(timeoutTask, 2000, 2000);
225 
226                 } catch (DocumentException e) {
227                     try {
228                         socket.close();
229                     }
230                     catch (IOException ioe) {
231                         // Do nothing
232                     }
233                     throw new ComponentException(e);
234                 } catch (XmlPullParserException e) {
235                     try {
236                         socket.close();
237                     }
238                     catch (IOException ioe) {
239                         // Do nothing
240                     }
241                     throw new ComponentException(e);
242                 }
243             } catch (XmlPullParserException e) {
244                 try {
245                     socket.close();
246                 }
247                 catch (IOException ioe) {
248                     // Do nothing
249                 }
250                 throw new ComponentException(e);
251             }
252         }
253         catch (UnknownHostException uhe) {
254             try {
255                 if (socket != null) socket.close();
256             }
257             catch (IOException e) {
258                 // Do nothing
259             }
260             throw new ComponentException(uhe);
261         }
262         catch (IOException ioe) {
263             try {
264                 if (socket != null) socket.close();
265             }
266             catch (IOException e) {
267                 // Do nothing
268             }
269             throw new ComponentException(ioe);
270         }
271     }

314     public void processPacket(final Packet packet) {
315         threadPool.execute(new Runnable() {
316             public void run() {
317                 if (packet instanceof IQ) {
318                     IQ iq = (IQ) packet;
319                     IQ.Type iqType = iq.getType();
320                     if (IQ.Type.result == iqType || IQ.Type.error == iqType) {
321                         // The server got an answer to an IQ packet that was sent from the component
322                         IQResultListener iqResultListener = resultListeners.remove(iq.getID());
323                         resultTimeout.remove(iq.getID());
324                         if (iqResultListener != null) {
325                             try {
326                                 iqResultListener.receivedAnswer(iq);
327                             }
328                             catch (Exception e) {
329                                  manager.getLog().error("Error processing answer of remote entity", e);
330                             }
331                             return;
332                         }
333                     }
334                 }
335                 component.processPacket(packet);
336             }
337         });
338     }

340     public void send(Packet packet) {
341         synchronized (writer) {
342             try {
343                 xmlSerializer.write(packet.getElement());
344                 xmlSerializer.flush();
345                 // Keep track of the last time a stanza was sent to the server
346                 lastActive = System.currentTimeMillis();
347             }
348             catch (IOException e) {
349                 // Log the exception
350                 manager.getLog().error(e.getMessage());
351                 if (!shutdown) {
352                     // Connection was lost so try to reconnect
353                     connectionLost();
354                 }
355             }
356         }
357     }

363     public void start() {
364         // Everything went fine so start reading packets from the server
365         readerThread = new SocketReadThread(this, reader);
366         readerThread.setDaemon(true);
367         readerThread.start();
368         // Notify the component that it will be notified of new received packets
369         component.start();
370     }

372     public void shutdown() {
373         shutdown = true;
374         disconnect();
375     }

377     private void disconnect() {
378         if (readerThread != null) {
379             readerThread.shutdown();
380         }
381         threadPool.shutdown();
382         TaskEngine.getInstance().cancelScheduledTask(keepAliveTask);
383         TaskEngine.getInstance().cancelScheduledTask(timeoutTask);
384         if (socket != null && !socket.isClosed()) {
385             try {
386                 synchronized (writer) {
387                     try {
388                         writer.write("</stream:stream>");
389                         xmlSerializer.flush();
390                     }
391                     catch (IOException e) {
392                         // Do nothing
393                     }
394                 }
395             }
396             catch (Exception e) {
397                 // Do nothing
398             }
399             try {
400                 socket.close();
401             }
402             catch (Exception e) {
403                 manager.getLog().error(e.getMessage());
404             }
405         }
406     }

408     /**
409      * Notification message that the connection with the server was lost unexpectedly. We will try
410      * to reestablish the connection for ever until the connection has been reestablished or this
411      * thread has been stopped.
412      */
413     public void connectionLost() {
414         // Ensure that only one thread will try to reconnect.
415         synchronized(this) {
416             if (reconnecting) {
417                 return;
418             }
419             reconnecting = true;
420         }
421         readerThread = null;
422         boolean isConnected = false;
423         if (!shutdown) {
424             // Notify the component that connection was lost so it needs to shutdown. The component is
425             // still registered in the local component manager but just not connected to the server
426             component.shutdown();
427         }
428         while (!isConnected && !shutdown) {
429             try {
430                 connect(host, port, subdomain);
431                 isConnected = true;
432                 // It may be possible that while a new connection was being established the
433                 // component was required to shutdown so in this case we need to close the new
434                 // connection
435                 if (shutdown) {
436                     disconnect();
437                 }
438                 else {
439                     // Component is back again working so start it up again
440                     start();
441                 }
442             } catch (ComponentException e) {
443                 manager.getLog().error("Error trying to reconnect with the server", e);
444                 // Wait for 5 seconds until the next retry
445                 try {
446                     Thread.sleep(5000);
447                 }
448                 catch (InterruptedException e1) {
449                     // Do nothing
450                 }
451             }
452         }
453         reconnecting = false;
454     }

456     /**
457      * Adds an {@link IQResultListener} that will be invoked when an IQ result is sent to the
458      * server itself and is of type result or error. This is a nice way for the server to
459      * send IQ packets to other XMPP entities and be waked up when a response is received back.<p>
460      *
461      * Once an IQ result was received, the listener will be invoked and removed from
462      * the list of listeners.
463      *
464      * @param id the id of the IQ packet being sent from the server to an XMPP entity.
465      * @param listener the IQResultListener that will be invoked when an answer is received
466      * @param timeoutmillis The amount of milliseconds after which waiting for a response should be stopped.
467      */
468     void addIQResultListener(String id, IQResultListener listener, long timeoutmillis) {
469         // be generated by the server and simulate like the client sent it. This will let listeners
470         // react and be removed from the collection
471         resultListeners.put(id, listener);
472         resultTimeout.put(id, System.currentTimeMillis() + timeoutmillis);
473     }

475     /**
476      * A TimerTask that keeps connections to the server alive by sending a space
477      * character on an interval.
478      */
479     private class KeepAliveTask extends TimerTask {
480 
481         public void run() {
482             synchronized (writer) {
483                 // Send heartbeat if no packet has been sent to the server for a given time
484                 if (System.currentTimeMillis() - lastActive >= 30000) {
485                     try {
486                         writer.write(" ");
487                         writer.flush();
488                     }
489                     catch (IOException e) {
490                         // Log the exception
491                         manager.getLog().error(e.getMessage());
492                         if (!shutdown) {
493                             // Connection was lost so try to reconnect
494                             connectionLost();
495                         }
496                     }
497                     catch (Exception e) {
498                         // Do nothing
499                     }
500                 }
501             }
502         }
503     }

505     /**
506 	 * Timer task that will remove Listeners that wait for results to IQ stanzas
507 	 * that have timed out. Time out values can be set to each listener
508 	 * individually by adjusting the timeout value in the third parameter of
509 	 * {@link ExternalComponent#addIQResultListener(String, IQResultListener, long)}.
510 	 *
511 	 * @author Guus der Kinderen, guus@nimbuzz.com
512 	 */
513     private class TimeoutTask extends TimerTask {
514 
515         /**
516          * Iterates over and removes all timed out results.<p>
517          *
518          * The map that keeps track of timeout values is ordered by timeout
519          * date. This way, iteration can be stopped as soon as the first value
520          * has been found that didn't timeout yet.
521          */
522         @Override
523         public void run() {
524             // Use an Iterator to allow changes to the Map that is backing
525             // the Iterator.
526             final Iterator<Map.Entry<String, Long>> it = resultTimeout.entrySet().iterator();
527 
528             while (it.hasNext()) {
529                 final Map.Entry<String, Long> pointer = it.next();
530 
531                 if (System.currentTimeMillis() < pointer.getValue()) {
532                     // This entry has not expired yet. Ignore it.
533                     continue;
534                 }
535 
536                 final String packetId = pointer.getKey();
537 
538                 // remove this listener from the list
539                 final IQResultListener listener = resultListeners.remove(packetId);
540                 if (listener != null) {
541                     // notify listener of the timeout.
542                     listener.answerTimeout(packetId);
543                 }
544 
545                 // remove the packet from the list that's used to track
546                 // timeouts
547                 it.remove();
548             }
549         }
