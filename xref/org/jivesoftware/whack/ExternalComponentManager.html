43  /**
44   * Implementation of the ComponentManager interface for external components.
45   * This implementation follows JEP-0014.
46   *
47   * @author Matt Tucker
48   */
49  public class ExternalComponentManager implements ComponentManager {
50  
51      /**
52       * Keeps the IP address or hostname of the server. This value will be used only for creating
53       * connections.
54       */
55      private String host;
56      /**
57       * Port of the server used for establishing new connections.
58       */
59      private int port;
60      /**
61       * Keeps the domain of the XMPP server. The domain may or may not match the host. The domain
62       * will be used mainly for the XMPP packets while the host is used mainly for creating
63       * connections to the server.
64       */
65      private String domain;
66      /**
67       * Timeout to use when trying to connect to the server.
68       */
69      private int connectTimeout = 2000;
70      /**
71       * This is a global secret key that will be used during the handshake with the server. If a
72       * secret key was not defined for the specific component then the global secret key will be
73       * used.
74       */
75      private String defaultSecretKey;
76      /**
77       * Keeps the secret keys to use for each subdomain. If a key was not found for a specific
78       * subdomain then the global secret key will used for the handshake with the server.
79       */
80      private Map<String, String> secretKeys = new Hashtable<String,String>();
81      /**
82       * Holds the settings for whether we will tell the XMPP server that a given component can connect
83       * to the same JID multiple times.  This is a custom Openfire extension and will not work
84       * with any other XMPP server. Other servers should ignore this setting.
85       */
86      private Map<String, Boolean> allowMultiple = new Hashtable<String,Boolean>();
87  
88      Preferences preferences = Preferences.userRoot();
89      private String preferencesPrefix;

91      /**
92       * Keeps a map that associates a domain with the external component thas is handling the domain.
93       */
94      private Map<String, ExternalComponent> componentsByDomain = new Hashtable<String,ExternalComponent>();
95      /**
96       * Keeps a map that associates a component with the wrapping ExternalComponent.
97       */
98      private Map<Component, ExternalComponent> components  = new Hashtable<Component,ExternalComponent>();
99  
100     private Logger logger = LoggerFactory.getLogger(ExternalComponentManager.class);;

102     /**
103      * Constructs a new ExternalComponentManager that will make connections
104      * to the specified XMPP server on the default port (5222).
105      *
106      * @param host the IP address or name of the XMPP server to connect to (e.g. "example.com").
107      */
108     public ExternalComponentManager(String host) {
109         this(host, 5225);
110     }
111 
112     /**
113      * Constructs a new ExternalComponentManager that will make connections to
114      * the specified XMPP server on the given port.
115      *
116      * @param host the IP address or name of the XMPP server to connect to (e.g. "example.com").
117      * @param port the port to connect on.
118      */
119     public ExternalComponentManager(String host, int port) {
120         if (host == null) {
121             throw new IllegalArgumentException("Host of XMPP server cannot be null");
122         }
123         this.host = host;
124         this.port = port;
125 
126         // Set this ComponentManager as the current component manager
127         ComponentManagerFactory.setComponentManager(this);
128     }
129 
130     /**
131      * Sets a secret key for a sub-domain, for future use by a component
132      * connecting to the server. Keys are used as an authentication mechanism
133      * when connecting to the server. Some servers may require a different
134      * key for each component, while others may use a global secret key.
135      *
136      * @param subdomain the sub-domain.
137      * @param secretKey the secret key
138      */
139     public void setSecretKey(String subdomain, String secretKey) {
140         secretKeys.put(subdomain, secretKey);
141     }
142 
143     /**
144      * Returns the secret key for a sub-domain. If no key was found then the default secret key
145      * will be returned.
146      *
147      * @param subdomain the subdomain to return its secret key.
148      * @return the secret key for a sub-domain.
149      */
150     public String getSecretKey(String subdomain) {
151         // Find the proper secret key to connect as the subdomain.
152         String secretKey = secretKeys.get(subdomain);
153         if (secretKey == null) {
154             secretKey = defaultSecretKey;
155         }
156         return secretKey;
157     }
158 
159     /**
160      * Sets the default secret key, which will be used when connecting if a
161      * specific secret key for the component hasn't been sent. Keys are used
162      * as an authentication mechanism when connecting to the server. Some servers
163      * may require a different key for each component, while others may use
164      * a global secret key.
165      *
166      * @param secretKey the default secret key.
167      */
168     public void setDefaultSecretKey(String secretKey) {
169         this.defaultSecretKey = secretKey;
170     }
171 
172     /**
173      * Returns if we want components to be able to connect multiple times to the same JID.  This is a custom
174      * Openfire extension and will not work with any other XMPP server. Other XMPP servers should ignore
175      * this extra setting.
176      *
177      * @param subdomain the sub-domain.
178      * @return True or false if we are allowing multiple connections.
179      */
180     public boolean isMultipleAllowed(String subdomain) {
181         Boolean allowed = allowMultiple.get(subdomain);
182         return allowed != null && allowed;
183     }
184 
185     /**
186      * Sets whether we will tell the XMPP server that we want multiple components to be able to connect
187      * to the same JID.  This is a custom Openfire extension and will not work with any other XMPP server.
188      * Other XMPP servers should ignore this extra setting.
189      *
190      * @param subdomain the sub-domain.
191      * @param allowMultiple Set to true if we want to allow multiple connections to same JID.
192      */
193     public void setMultipleAllowed(String subdomain, boolean allowMultiple) {
194         this.allowMultiple.put(subdomain, allowMultiple);
195     }

197 	@Override
198     public void addComponent(String subdomain, Component component) throws ComponentException {
199         addComponent(subdomain, component, this.port);
200     }
201 
202     public void addComponent(String subdomain, Component component, int port) throws ComponentException {
203         if (componentsByDomain.containsKey(subdomain)) {
204             if (componentsByDomain.get(subdomain).getComponent() == component) {
205                 // Do nothing since the component has already been registered
206                 return;
207             }
208             else {
209                 throw new IllegalArgumentException("Subdomain already in use by another component");
210             }
211         }
212         // Create a wrapping ExternalComponent on the component
213         ExternalComponent externalComponent = new ExternalComponent(component, this);
214         try {
215             // Register the new component
216             componentsByDomain.put(subdomain, externalComponent);
217             components.put(component, externalComponent);
218             // Ask the ExternalComponent to connect with the remote server
219             externalComponent.connect(host, port, subdomain);
220             // Initialize the component
221             JID componentJID = new JID(null, externalComponent.getDomain(), null);
222             externalComponent.initialize(componentJID, this);
223         } catch (ComponentException e) {
224             // Unregister the new component
225             componentsByDomain.remove(subdomain);
226             components.remove(component);
227             // Re-throw the exception
228             throw e;
229         }
230         // Ask the external component to start processing incoming packets
231         externalComponent.start();
232     }
233 
234     public void removeComponent(String subdomain) throws ComponentException {
235         ExternalComponent externalComponent = componentsByDomain.remove(subdomain);
236         if (externalComponent != null) {
237             components.remove(externalComponent.getComponent());
238             externalComponent.shutdown();
239         }
240     }

242 	@Override
243   public void sendPacket(Component component, Packet packet) {
244         // Get the ExternalComponent that is wrapping the specified component and ask it to
245         // send the packet
246         components.get(component).send(packet);
247     }
248 
249 	@Override
250    public IQ query(Component component, IQ packet, long timeout) throws ComponentException {
251         final LinkedBlockingQueue<IQ> answer = new LinkedBlockingQueue<IQ>(8);
252         ExternalComponent externalComponent = components.get(component);
253         externalComponent.addIQResultListener(packet.getID(), new IQResultListener() {
254             public void receivedAnswer(IQ packet) {
255                 answer.offer(packet);
256             }
257 
258             public void answerTimeout(String packetId) {
259                 //Do nothing
260             }
261         }, timeout);
262         sendPacket(component, packet);
263         IQ reply = null;
264         try {
265             reply = answer.poll(timeout, TimeUnit.MILLISECONDS);
266         } catch (InterruptedException e) {
267             // Ignore
268         }
269         return reply;
270     }
271 
272 	@Override
273    public void query(Component component, IQ packet, IQResultListener listener) throws ComponentException {
274         ExternalComponent externalComponent = components.get(component);
275         // Add listenet with a timeout of 5 minutes to prevent memory leaks
276         externalComponent.addIQResultListener(packet.getID(), listener, 300000);
277         sendPacket(component, packet);
278     }

281     public String getProperty(String name) {
282         return preferences.get(getPreferencesPrefix() + name, null);
283     }
284 
285     public void setProperty(String name, String value) {
286         preferences.put(getPreferencesPrefix() + name, value);
287     }
288 
289     private String getPreferencesPrefix() {
290         if (preferencesPrefix == null) {
291             preferencesPrefix = "whack." + domain + ".";
292         }
293         return preferencesPrefix;
294     }

298      * will be used mainly for the XMPP packets while the host is used mainly for creating
299      * connections to the server.
300      *
301      * @param domain the domain of the XMPP server.
302      */
303     public void setServerName(String domain) {
304         this.domain = domain;
305     }

307     /**
308      * Returns the domain of the XMPP server where we are connected to or <tt>null</tt> if
309      * this value was never configured. When the value is null then the component will
310      * register with just its subdomain and we expect the server to accept the component and
311      * append its domain to form the JID of the component.
312      *
313      * @return the domain of the XMPP server or null if never configured.
314      */
315     public String getServerName() {
316         return domain;
317     }
318 
319     /**
320      * Returns the timeout (in milliseconds) to use when trying to connect to the server.
321      * The default value is 2 seconds.
322      *
323      * @return the timeout to use when trying to connect to the server.
324      */
325     public int getConnectTimeout() {
326         return connectTimeout;
327     }
328 
329     /**
330      * Sets the timeout (in milliseconds) to use when trying to connect to the server.
331      * The default value is 2 seconds.
332      *
333      * @param connectTimeout the timeout, in milliseconds, to use when trying to connect to the server.
334      */
335     public void setConnectTimeout(int connectTimeout) {
336         this.connectTimeout = connectTimeout;
337     }

341     }

344         return logger;
