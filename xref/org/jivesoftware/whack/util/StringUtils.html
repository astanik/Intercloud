50      /**
51       * Returns the name portion of a XMPP address. For example, for the
52       * address "matt@jivesoftware.com/Smack", "matt" would be returned. If no
53       * username is present in the address, the empty string will be returned.
54       *
55       * @param XMPPAddress the XMPP address.
56       * @return the name portion of the XMPP address.
57       */
58      public static String parseName(String XMPPAddress) {
59          if (XMPPAddress == null) {
60              return null;
61          }
62          int atIndex = XMPPAddress.indexOf("@");
63          if (atIndex <= 0) {
64              return "";
65          }
66          else {
67              return XMPPAddress.substring(0, atIndex);
68          }
69      }

71      /**
72       * Returns the server portion of a XMPP address. For example, for the
73       * address "matt@jivesoftware.com/Smack", "jivesoftware.com" would be returned.
74       * If no server is present in the address, the empty string will be returned.
75       *
76       * @param XMPPAddress the XMPP address.
77       * @return the server portion of the XMPP address.
78       */
79      public static String parseServer(String XMPPAddress) {
80          if (XMPPAddress == null) {
81              return null;
82          }
83          int atIndex = XMPPAddress.indexOf("@");
84          // If the String ends with '@', return the empty string.
85          if (atIndex + 1 > XMPPAddress.length()) {
86              return "";
87          }
88          int slashIndex = XMPPAddress.indexOf("/");
89          if (slashIndex > 0) {
90              return XMPPAddress.substring(atIndex + 1, slashIndex);
91          }
92          else {
93              return XMPPAddress.substring(atIndex + 1);
94          }
95      }

97      /**
98       * Returns the resource portion of a XMPP address. For example, for the
99       * address "matt@jivesoftware.com/Smack", "Smack" would be returned. If no
100      * resource is present in the address, the empty string will be returned.
101      *
102      * @param XMPPAddress the XMPP address.
103      * @return the resource portion of the XMPP address.
104      */
105     public static String parseResource(String XMPPAddress) {
106         if (XMPPAddress == null) {
107             return null;
108         }
109         int slashIndex = XMPPAddress.indexOf("/");
110         if (slashIndex + 1 > XMPPAddress.length() || slashIndex < 0) {
111             return "";
112         }
113         else {
114             return XMPPAddress.substring(slashIndex + 1);
115         }
116     }

118     /**
119      * Returns the XMPP address with any resource information removed. For example,
120      * for the address "matt@jivesoftware.com/Smack", "matt@jivesoftware.com" would
121      * be returned.
122      *
123      * @param XMPPAddress the XMPP address.
124      * @return the bare XMPP address without resource information.
125      */
126     public static String parseBareAddress(String XMPPAddress) {
127         if (XMPPAddress == null) {
128             return null;
129         }
130         int slashIndex = XMPPAddress.indexOf("/");
131         if (slashIndex < 0) {
132             return XMPPAddress;
133         }
134         else if (slashIndex == 0) {
135             return "";
136         }
137         else {
138             return XMPPAddress.substring(0, slashIndex);
139         }
140     }

142     /**
143      * Escapes all necessary characters in the String so that it can be used
144      * in an XML doc.
145      *
146      * @param string the string to escape.
147      * @return the string with appropriate characters escaped.
148      */
149     public static final String escapeForXML(String string) {
150         if (string == null) {
151             return null;
152         }
153         char ch;
154         int i=0;
155         int last=0;
156         char[] input = string.toCharArray();
157         int len = input.length;
158         StringBuffer out = new StringBuffer((int)(len*1.3));
159         for (; i < len; i++) {
160             ch = input[i];
161             if (ch > '>') {
162                 continue;
163             }
164             else if (ch == '<') {
165                 if (i > last) {
166                     out.append(input, last, i - last);
167                 }
168                 last = i + 1;
169                 out.append(LT_ENCODE);
170             }
171             else if (ch == '>') {
172                 if (i > last) {
173                     out.append(input, last, i - last);
174                 }
175                 last = i + 1;
176                 out.append(GT_ENCODE);
177             }
178 
179             else if (ch == '&') {
180                 if (i > last) {
181                     out.append(input, last, i - last);
182                 }
183                 // Do nothing if the string is of the form &#235; (unicode value)
184                 if (!(len > i + 5
185                     && input[i + 1] == '#'
186                     && Character.isDigit(input[i + 2])
187                     && Character.isDigit(input[i + 3])
188                     && Character.isDigit(input[i + 4])
189                     && input[i + 5] == ';')) {
190                         last = i + 1;
191                         out.append(AMP_ENCODE);
192                     }
193             }
194             else if (ch == '"') {
195                 if (i > last) {
196                     out.append(input, last, i - last);
197                 }
198                 last = i + 1;
199                 out.append(QUOTE_ENCODE);
200             }
201         }
202         if (last == 0) {
203             return string;
204         }
205         if (i > last) {
206             out.append(input, last, i - last);
207         }
208         return out.toString();
209     }

216     /**
217      * Hashes a String using the SHA-1 algorithm and returns the result as a
218      * String of hexadecimal numbers. This method is synchronized to avoid
219      * excessive MessageDigest object creation. If calling this method becomes
220      * a bottleneck in your code, you may wish to maintain a pool of
221      * MessageDigest objects instead of using this method.
222      * <p>
223      * A hash is a one-way function -- that is, given an
224      * input, an output is easily computed. However, given the output, the
225      * input is almost impossible to compute. This is useful for passwords
226      * since we can store the hash and a hacker will then have a very hard time
227      * determining the original password.
228      *
229      * @param data the String to compute the hash of.
230      * @return a hashed version of the passed-in String
231      */
232     public synchronized static final String hash(String data) {
233         if (digest == null) {
234             try {
235                 digest = MessageDigest.getInstance("SHA-1");
236             }
237             catch (NoSuchAlgorithmException nsae) {
238                 System.err.println("Failed to load the SHA-1 MessageDigest. " +
239                 "Jive will be unable to function normally.");
240             }
241         }
242         // Now, compute hash.
243         try {
244             digest.update(data.getBytes("UTF-8"));
245         }
246         catch (UnsupportedEncodingException e) {
247             System.err.println(e);
248         }
249         return encodeHex(digest.digest());
250     }

252     /**
253      * Turns an array of bytes into a String representing each byte as an
254      * unsigned hex number.
255      * <p>
256      * Method by Santeri Paavolainen, Helsinki Finland 1996<br>
257      * (c) Santeri Paavolainen, Helsinki Finland 1996<br>
258      * Distributed under LGPL.
259      *
260      * @param bytes an array of bytes to convert to a hex-string
261      * @return generated hex string
262      */
263     public static final String encodeHex(byte[] bytes) {
264         StringBuffer buf = new StringBuffer(bytes.length * 2);
265         int i;
266 
267         for (i = 0; i < bytes.length; i++) {
268             if (((int) bytes[i] & 0xff) < 0x10) {
269                 buf.append("0");
270             }
271             buf.append(Long.toString((int) bytes[i] & 0xff, 16));
272         }
273         return buf.toString();
274     }

290     /**
291      * Encodes a String as a base64 String.
292      *
293      * @param data a String to encode.
294      * @return a base64 encoded String.
295      */
296     public static String encodeBase64(String data) {
297         byte[] bytes = null;
298         try {
299             bytes = data.getBytes("UTF-8");
300         }
301         catch (UnsupportedEncodingException uee) {
302             uee.printStackTrace();
303         }
304         return encodeBase64(bytes);
305     }

307     /**
308      * Encodes a byte array into a base64 String.
309      *
310      * @param data a byte array to encode.
311      * @return a base64 encode String.
312      */
313     public static String encodeBase64(byte[] data) {
314         // Encode the String. We pass in a flag to specify that line
315         // breaks not be added. This is consistent with our previous base64
316         // implementation. Section 2.1 of 3548 (base64 spec) also specifies
317         // no line breaks by default.
318         return Base64.encodeBytes(data, Base64.DONT_BREAK_LINES);
319     }

338     /**
339      * Array of numbers and letters of mixed case. Numbers appear in the list
340      * twice so that there is a more equal chance that a number will be picked.
341      * We can use the array to get a random number or letter by picking a random
342      * array index.
343      */
344     private static char[] numbersAndLetters = ("0123456789abcdefghijklmnopqrstuvwxyz" +
345                     "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ").toCharArray();
346 
347     /**
348      * Returns a random String of numbers and letters (lower and upper case)
349      * of the specified length. The method uses the Random class that is
350      * built-in to Java which is suitable for low to medium grade security uses.
351      * This means that the output is only pseudo random, i.e., each number is
352      * mathematically generated so is not truly random.<p>
353      *
354      * The specified length must be at least one. If not, the method will return
355      * null.
356      *
357      * @param length the desired length of the random String to return.
358      * @return a random String of numbers and letters of the specified length.
359      */
360     public static final String randomString(int length) {
361         if (length < 1) {
362             return null;
363         }
364         // Create a char buffer to put random letters and numbers in.
365         char [] randBuffer = new char[length];
366         for (int i=0; i<randBuffer.length; i++) {
367             randBuffer[i] = numbersAndLetters[randGen.nextInt(71)];
368         }
369         return new String(randBuffer);
370     }
