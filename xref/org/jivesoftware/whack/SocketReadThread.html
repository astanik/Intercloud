21  package org.jivesoftware.whack;
22  
23  import org.dom4j.Element;
24  import org.dom4j.io.XPPPacketReader;
25  import org.xmlpull.v1.XmlPullParserException;
26  
27  import de.tu_berlin.cit.intercloud.xmpp.core.packet.IQ;
28  import de.tu_berlin.cit.intercloud.xmpp.core.packet.Message;
29  import de.tu_berlin.cit.intercloud.xmpp.core.packet.Packet;
30  import de.tu_berlin.cit.intercloud.xmpp.core.packet.Presence;
31  import de.tu_berlin.cit.intercloud.xmpp.core.packet.Roster;
32  
33  import java.io.EOFException;
34  import java.net.SocketException;
35  
36  /**
37   * Reads XMPP XML packets from a socket and asks the component to process the packets.
38   *
39   * @author Gaston Dombiak
40   */
41  class SocketReadThread extends Thread {
42  
43      private ExternalComponent component;
44      private boolean shutdown = false;
45  
46      XPPPacketReader reader = null;
47  
48      /**
49       * Create dedicated read thread for this socket.
50       *
51       * @param component  The component for which this thread is reading for
52       * @param reader     The reader to use for reading
53       */
54      public SocketReadThread(ExternalComponent component, XPPPacketReader reader) {
55          super("Component socket reader");
56          this.component = component;
57          this.reader = reader;
58      }
59  
60      /**
61       * A dedicated thread loop for reading the stream and sending incoming
62       * packets to the appropriate router.
63       */
64      public void run() {
65          try {
66              readStream();
67          }
68          catch (EOFException eof) {
69              // Normal disconnect
70          }
71          catch (SocketException se) {
72              // Do nothing if the exception occured while shutting down the component otherwise
73              // log the error and try to establish a new connection
74              if (!shutdown) {
75                  component.getManager().getLog().error(se.getMessage());
76                  component.connectionLost();
77              }
78          }
79          catch (XmlPullParserException ie) {
80              component.getManager().getLog().error(ie.getMessage());
81          }
82          catch (Exception e) {
83              component.getManager().getLog().warn(e.getMessage());
84          }
85      }
86  
87      /**
88       * Read the incoming stream until it ends.
89       */
90      private void readStream() throws Exception {
91          while (!shutdown) {
92              Element doc = reader.parseDocument().getRootElement();
93  
94              if (doc == null) {
95                  // Stop reading the stream since the server has sent an end of stream element and
96                  // probably closed the connection
97                  return;
98              }
99  
100             Packet packet;
101             String tag = doc.getName();
102             if ("message".equals(tag)) {
103                 packet = new Message(doc);
104             }
105             else if ("presence".equals(tag)) {
106                 packet = new Presence(doc);
107             }
108             else if ("iq".equals(tag)) {
109                 packet = getIQ(doc);
110             }
111             else {
112                 throw new XmlPullParserException("Unknown packet type was read: " + tag);
113             }
114             // Request the component to process the received packet
115             component.processPacket(packet);
116         }
117     }
118 
119     private IQ getIQ(Element doc) {
120         Element query = doc.element("query");
121         if (query != null && "jabber:iq:roster".equals(query.getNamespaceURI())) {
122             return new Roster(doc);
123         }
124         else {
125             return new IQ(doc);
126         }
127     }
128 
129     /**
130      * Aks the thread to stop reading packets. The thread may not stop immediatelly so if a socket
131      * exception occurs because the connection was lost then no exception will be logged nor the
132      * component will try to reestablish the connection.<p>
133      *
134      * Once this method was sent this instance should be discarded and created a new one if a new
135      * connection with the server is established.
136      */
137     public void shutdown() {
138         shutdown = true;
139     }
140 }
