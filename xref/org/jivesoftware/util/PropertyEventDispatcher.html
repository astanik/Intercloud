12  package org.jivesoftware.util;
13  
14  import java.util.List;
15  import java.util.Map;
16  import java.util.concurrent.CopyOnWriteArrayList;
17  
18  /**
19   * Dispatches property events. Each event has a {@link EventType type}
20   * and optional parameters, as follows:<p>
21   *
22   * <table border="1">
23   * <tr><th>Event Type</th><th>Extra Params</th></tr>
24   * <tr><td>{@link EventType#property_set property_set}</td><td>A param named <tt>value</tt> that
25   *      has the value of the property set.</td></tr>
26   * <tr><td>{@link EventType#property_deleted property_deleted}</td><td><i>None</i></td></tr>
27   * <tr><td>{@link EventType#xml_property_set xml_property_set}</td><td>A param named <tt>value</tt> that
28   *      has the value of the property set.</td></tr>
29   * <tr><td>{@link EventType#xml_property_deleted xml_property_deleted}</td><td><i>None</i></td></tr>
30   * </table>
31   *
32   * @author Matt Tucker
33   */
34  public class PropertyEventDispatcher {
35  
36      private static List<PropertyEventListener> listeners =
37              new CopyOnWriteArrayList<PropertyEventListener>();
38  
39      private PropertyEventDispatcher() {
40          // Not instantiable.
41      }
42  
43      /**
44       * Registers a listener to receive events.
45       *
46       * @param listener the listener.
47       */
48      public static void addListener(PropertyEventListener listener) {
49          if (listener == null) {
50              throw new NullPointerException();
51          }
52          listeners.add(listener);
53      }
54  
55      /**
56       * Unregisters a listener to receive events.
57       *
58       * @param listener the listener.
59       */
60      public static void removeListener(PropertyEventListener listener) {
61          listeners.remove(listener);
62      }
63  
64      /**
65       * Dispatches an event to all listeners.
66       *
67       * @param property the property.
68       * @param eventType the event type.
69       * @param params event parameters.
70       */
71      public static void dispatchEvent(String property, EventType eventType, Map params) {
72          for (PropertyEventListener listener : listeners) {
73              try {
74                  switch (eventType) {
75                      case property_set: {
76                          listener.propertySet(property, params);
77                          break;
78                      }
79                      case property_deleted: {
80                          listener.propertyDeleted(property, params);
81                          break;
82                      }
83                      case xml_property_set: {
84                          listener.xmlPropertySet(property, params);
85                          break;
86                      }
87                      case xml_property_deleted: {
88                          listener.xmlPropertyDeleted(property, params);
89                          break;
90                      }
91                      default:
92                          break;
93                  }
94              }
95              catch (Exception e) {
96                  e.printStackTrace();
97              }
98          }
99      }
100 
101     /**
102      * Represents valid event types.
103      */
104     public enum EventType {
105 
106         /**
107          * A property was set.
108          */
109         property_set,
110 
111         /**
112          * A property was deleted.
113          */
114         property_deleted,
115 
116         /**
117          * An XML property was set.
118          */
119         xml_property_set,
120 
121         /**
122          * An XML property was deleted.
123          */
124         xml_property_deleted
125     }
126 }
